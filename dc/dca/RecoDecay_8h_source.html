13 float energy(float px, float py, float pz, float mass)
14 {
15  float en_ = sqrtf(mass * mass + px * px + py * py + pz * pz);
16  return en_;
17 };

19 float invmass2prongs2(float px0, float py0, float pz0, float mass0,
20  float px1, float py1, float pz1, float mass1)
21 {
22 
23  float energy0_ = energy(px0, py0, pz0, mass0);
24  float energy1_ = energy(px1, py1, pz1, mass1);
25  float energytot = energy0_ + energy1_;
26 
27  float psum2 = (px0 + px1) * (px0 + px1) +
28  (py0 + py1) * (py0 + py1) +
29  (pz0 + pz1) * (pz0 + pz1);
30  return energytot * energytot - psum2;
31 };

33 float invmass3prongs2(float px0, float py0, float pz0, float mass0,
34  float px1, float py1, float pz1, float mass1,
35  float px2, float py2, float pz2, float mass2)
36 {

38  float energy1_ = energy(px1, py1, pz1, mass1);
39  float energy2_ = energy(px2, py2, pz2, mass2);
40  float energytot = energy0_ + energy1_ + energy2_;
41 
42  float psum2 = (px0 + px1 + px2) * (px0 + px1 + px2) +
43  (py0 + py1 + py2) * (py0 + py1 + py2) +
44  (pz0 + pz1 + pz2) * (pz0 + pz1 + pz2);
45  return energytot * energytot - psum2;
46 };
