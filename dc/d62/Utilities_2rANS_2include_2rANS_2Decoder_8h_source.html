94 template <typename coder_T, typename stream_T, typename source_T>
95 template <typename stream_IT, typename source_IT>
96 void Decoder<coder_T, stream_T, source_T>::process(const source_IT outputBegin, const stream_IT inputEnd, size_t numSymbols) const
97 {
98  LOG(trace) << "start decoding";
99  RANSTimer t;
100  t.start();
101  static_assert(std::is_same<typename std::iterator_traits<source_IT>::value_type, source_T>::value);
102  static_assert(std::is_same<typename std::iterator_traits<stream_IT>::value_type, stream_T>::value);
103 
104  if (numSymbols == 0) {
105  LOG(warning) << "Empty message passed to decoder, skipping decode process";
106  return;
107  }
108 
109  ransDecoder rans0, rans1;
110  stream_IT inputIter = inputEnd;
111  source_IT it = outputBegin;
112 
113  // make Iter point to the last last element
114  --inputIter;
115 
116  inputIter = rans0.decInit(inputIter);
117  inputIter = rans1.decInit(inputIter);
118 
119  for (size_t i = 0; i < (numSymbols & ~1); i += 2) {
120  const stream_T s0 =
121  (*mReverseLUT)[rans0.decGet(mProbabilityBits)];
122  const stream_T s1 =
123  (*mReverseLUT)[rans1.decGet(mProbabilityBits)];
124  *it++ = s0;
125  *it++ = s1;
126  rans0.decAdvanceSymbolStep((*mSymbolTable)[s0], mProbabilityBits);
127  rans1.decAdvanceSymbolStep((*mSymbolTable)[s1], mProbabilityBits);
128  inputIter = rans0.decRenorm(inputIter);
129  inputIter = rans1.decRenorm(inputIter);
130  }
131 
132  // last byte, if number of bytes was odd
133  if (numSymbols & 1) {
134  const stream_T s0 =
135  (*mReverseLUT)[rans0.decGet(mProbabilityBits)];
136  *it = s0;
137  inputIter = rans0.decAdvanceSymbol(inputIter, (*mSymbolTable)[s0], mProbabilityBits);
138  }
139  t.stop();
140  LOG(debug1) << __func__ << " inclusive time (ms): " << t.getDurationMS();
141 
142  LOG(trace) << "done decoding";
143 }
