38  const dataformats::MCTruthContainer<MCCompLabel>* mcLabels)

44  auto first = rof.getFirstEntry();
45  auto clusters_in_frame = rof.getROFData(clusters);

56  locXYZ = dict.getClusterCoordinates(c);
57  } else {
58  o2::itsmft::ClusterPattern patt(pattIt);
59  locXYZ = dict.getClusterCoordinates(c, patt);
60  }
61  } else {
62  o2::itsmft::ClusterPattern patt(pattIt);
63  locXYZ = dict.getClusterCoordinates(c, patt);
64  }
65  auto sensorID = c.getSensorID();

67  auto gloXYZ = geom->getMatrixL2G(sensorID) * locXYZ;
68

71  Float_t phiCoord = clsPoint2D.Phi();

73  int rBinIndex = constants::index_table::getRBinIndex(rCoord);
74  int phiBinIndex = constants::index_table::getPhiBinIndex(phiCoord);
75  int binIndex = constants::index_table::getBinIndex(rBinIndex, phiBinIndex);
76 
77  event.addClusterToLayer(layer, gloXYZ.x(), gloXYZ.y(), gloXYZ.z(), phiCoord, rCoord, event.getClustersInLayer(layer).size(), binIndex);
78  if (mcLabels) {
79  event.addClusterLabelToLayer(layer, *(mcLabels->getLabels(first + clusterId).begin()));
80  }
81  event.addClusterExternalIndexToLayer(layer, first + clusterId);
82  clusterId++;
83  }
84  return clusters_in_frame.size();
85 }
