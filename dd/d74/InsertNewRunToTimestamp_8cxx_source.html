57 int main(int argc, char* argv[])
58 {
59  bpo::options_description options("Allowed options");

65  o2::ccdb::CcdbApi api;
66  const std::string url = vm["url"].as<std::string>();
67  api.init(url);
68  if (!api.isHostReachable()) {
69  LOG(WARNING) << "CCDB host " << url << " is not reacheable, cannot go forward";
70  return 1;
71  }
72 
73  std::string path = vm["path"].as<std::string>();
74  std::map<std::string, std::string> metadata;
75  std::map<std::string, std::string>* headers;
76  long start = vm["start"].as<long>();
77  long stop = vm["stop"].as<long>();
78 
79  RunToTimestamp* converter = api.retrieveFromTFileAny<RunToTimestamp>(path, metadata, -1, headers);
80 
81  if (!converter) {
82  LOG(INFO) << "Did not retrieve run number to timestamp converter, creating a new one!";
83  converter = new RunToTimestamp();
84  } else {
85  LOG(INFO) << "Retrieved run number to timestamp converter from ccdb url" << url;
86  }
87 
88  if (vm["update"].as<int>())
89  converter->update(vm["run"].as<uint>(), vm["timestamp"].as<long>());
90  else
91  converter->insert(vm["run"].as<uint>(), vm["timestamp"].as<long>());
92 
93  if (vm["verbose"].as<int>())
94  converter->print();
95 
96  api.storeAsTFileAny(converter, path, metadata, start, stop);
97 
98  return 0;
99 }
