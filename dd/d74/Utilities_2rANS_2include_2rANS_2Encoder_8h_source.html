101 template <typename coder_T, typename stream_T, typename source_T>
102 template <typename stream_IT, typename source_IT>
103 const stream_IT Encoder<coder_T, stream_T, source_T>::Encoder::process(
104  const stream_IT outputBegin, const stream_IT outputEnd, const source_IT inputBegin, const source_IT inputEnd) const
105 {
106  LOG(trace) << "start encoding";
107  RANSTimer t;
108  t.start();
109 
110  static_assert(std::is_same<typename std::iterator_traits<source_IT>::value_type, source_T>::value);
111  static_assert(std::is_same<typename std::iterator_traits<stream_IT>::value_type, stream_T>::value);
112 
113  if (inputBegin == inputEnd) {
114  LOG(warning) << "passed empty message to encoder, skip encoding";
115  return outputEnd;
116  }
117 
118  if (outputBegin == outputEnd) {
119  const std::string errorMessage("Unallocated encode buffer passed to encoder. Aborting");
120  LOG(error) << errorMessage;
121  throw std::runtime_error(errorMessage);
122  }
123 
124  ransCoder rans0, rans1;
125  rans0.encInit();
126  rans1.encInit();
127 
128  stream_IT outputIter = outputBegin;
129  source_IT inputIT = inputEnd;
130 
131  const auto inputBufferSize = std::distance(inputBegin, inputEnd);
132 
133  // odd number of bytes?
134  if (inputBufferSize & 1) {
135  const coder_T s = *(--inputIT);
136  outputIter = rans0.encPutSymbol(outputIter, (*mSymbolTable)[s], mProbabilityBits);
137  assert(outputIter < outputEnd);
138  }
139 
140  while (inputIT > inputBegin) { // NB: working in reverse!
141  const coder_T s1 = *(--inputIT);
142  const coder_T s0 = *(--inputIT);
143  outputIter = rans1.encPutSymbol(outputIter, (*mSymbolTable)[s1], mProbabilityBits);
144  outputIter = rans0.encPutSymbol(outputIter, (*mSymbolTable)[s0], mProbabilityBits);
145  assert(outputIter < outputEnd);
146  }
147  outputIter = rans1.encFlush(outputIter);
148  outputIter = rans0.encFlush(outputIter);
149  // first iterator past the range so that sizes, distances and iterators work correctly.
150  ++outputIter;
151 
152  assert(!(outputIter > outputEnd));
153 
154  // deal with overflow
155  if (outputIter > outputEnd) {
156  const std::string exceptionText = [&]() {
157  std::stringstream ss;
158  ss << __func__ << " detected overflow in encode buffer: allocated:" << std::distance(outputBegin, outputEnd) << ", used:" << std::distance(outputBegin, outputIter);
159  return ss.str();
160  }();
161 
162  LOG(error) << exceptionText;
163  throw std::runtime_error(exceptionText);
164  }
165 
166  t.stop();
167  LOG(debug1) << __func__ << " inclusive time (ms): " << t.getDurationMS();
168 
169 // advanced diagnostics for debug builds
170 #if !defined(NDEBUG)
171  LOG(debug2) << "EncoderProperties: {"
172  << "sourceTypeB: " << sizeof(source_T) << ", "
173  << "streamTypeB: " << sizeof(stream_T) << ", "
174  << "coderTypeB: " << sizeof(coder_T) << ", "
175  << "probabilityBits: " << mProbabilityBits << ", "
176  << "inputBufferSizeB: " << inputBufferSize * sizeof(source_T) << ", "
177  << "outputBufferSizeB: " << std::distance(outputBegin, outputIter) * sizeof(stream_T) << "}";
178 #endif
179 
180  LOG(trace) << "done encoding";
181 
182  return outputIter;
183 };
