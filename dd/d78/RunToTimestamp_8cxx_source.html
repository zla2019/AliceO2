20 bool RunToTimestamp::insert(uint runNumber, long timestamp)
21 {
22  std::pair<std::map<uint, long>::iterator, bool> check;
23  check = mMap.insert(std::pair<uint, long>(runNumber, timestamp));
24  if (!check.second) {
25  LOG(FATAL) << "Run number " << runNumber << " already existed with a timestamp of " << check.first->second;
26  return false;
27  }
28  LOG(INFO) << "Add new run " << runNumber << " with timestamp " << timestamp << " to converter";
29  return true;

32 bool RunToTimestamp::update(uint runNumber, long timestamp)
33 {
34  if (!Has(runNumber)) {
35  LOG(FATAL) << "Run to Timestamp converter does not have run " << runNumber << ", cannot update converter";
36  return false;
37  }
38  mMap[runNumber] = timestamp;
39  return true;
