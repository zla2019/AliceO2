61 bool msgDigitSortComparator(o2::trd::Digit const& a, o2::trd::Digit const& b)
62 {
63  FeeParam* fee = FeeParam::instance();
64  int rowa = a.getRow();
65  int rowb = b.getRow();
66  int pada = a.getPad();
67  int padb = b.getPad();
68  double timea = a.getTimeStamp();
69  double timeb = b.getTimeStamp();
70  int roba = fee->getROBfromPad(rowa, pada);
71  int robb = fee->getROBfromPad(rowb, padb);
72  int mcma = fee->getMCMfromPad(rowa, pada);
73  int mcmb = fee->getMCMfromPad(rowb, padb);
74  //LOG(info) << "comparing " << rowa << ":" << pada <<":" << roba <<" "<< mcma << " with " << rowb << ":" << padb <<":" << robb <<" "<< mcmb;
75  if (timea < timeb) {
76  // LOG(info) << "yip timea < timeb " << timea <<"<" << timeb;
77  return 1;
78  } else if (timea == timeb) {
79 
80  if (a.getDetector() < b.getDetector())
81  return 1;
82  else {
83  if (a.getDetector() == b.getDetector()) {
84  if (roba < robb)
85  return 1;
86  else {
87  if (roba == robb) {
88  if (mcma < mcmb)
89  return 1;
90  else
91  return 0;
92  } else
93  return 0;
94  }
95  return 0;
96  }
97  return 0;
98  }
99  return 0;
100  }
101  return 0;
102 }

190 void TRDDPLTrapSimulatorTask::setOnlineGainTables()
191 {
192  const int nDets = 540; //TRDGeometry::Ndet();
193  const int nMcms = TRDGeometry::MCMmax();
194  const int nChs = TRDGeometry::ADCmax();
195  //check FGBY from trapconfig.
196  //check the input parameter of trd-onlinegaincorrection.
197  //warn if you have chosen a trapconfig with gaincorrections but chosen not to use them.
198  if (mEnableOnlineGainCorrection) {
199  if (mTrapConfig->getTrapReg(TrapConfig::kFGBY) == 0) {
200  LOG(warn) << "you have asked to do online gain calibrations but the selected trap config does not have FGBY enabled, so modifying trapconfig to conform to your command line request. OnlineGains will be 1, i.e. no effect.";
201  for (int iDet = 0; iDet < nDets; ++iDet) {
202  mTrapConfig->setTrapReg(TrapConfig::kFGBY, 1, iDet);
203  }
204  }
205  mCalib->setOnlineGainTables(mOnlineGainTableName);
206  //TODO add some error checking inhere.
207  // gain factors are per MCM
208  // allocate the registers accordingly
209  for (int ch = 0; ch < nChs; ++ch) {
210  TrapConfig::TrapReg_t regFGAN = (TrapConfig::TrapReg_t)(TrapConfig::kFGA0 + ch);
211  TrapConfig::TrapReg_t regFGFN = (TrapConfig::TrapReg_t)(TrapConfig::kFGF0 + ch);
212  mTrapConfig->setTrapRegAlloc(regFGAN, TrapConfig::kAllocByMCM);
213  mTrapConfig->setTrapRegAlloc(regFGFN, TrapConfig::kAllocByMCM);
214  }
215 
216  for (int iDet = 0; iDet < nDets; ++iDet) {
217  const int nRobs = TRDGeometry::getStack(iDet) == 2 ? TRDGeometry::ROBmaxC0() : TRDGeometry::ROBmaxC1();
218  for (int rob = 0; rob < nRobs; ++rob) {
219  for (int mcm = 0; mcm < nMcms; ++mcm) {
220  // set ADC reference voltage
221  mTrapConfig->setTrapReg(TrapConfig::kADCDAC, mCalib->getOnlineGainAdcdac(iDet, rob, mcm), iDet, rob, mcm);
222  // set constants channel-wise
223  for (int ch = 0; ch < nChs; ++ch) {
224  TrapConfig::TrapReg_t regFGAN = (TrapConfig::TrapReg_t)(TrapConfig::kFGA0 + ch);
225  TrapConfig::TrapReg_t regFGFN = (TrapConfig::TrapReg_t)(TrapConfig::kFGF0 + ch);
226  mTrapConfig->setTrapReg(regFGAN, mCalib->getOnlineGainFGAN(iDet, rob, mcm, ch), iDet, rob, mcm);
227  mTrapConfig->setTrapReg(regFGFN, mCalib->getOnlineGainFGFN(iDet, rob, mcm, ch), iDet, rob, mcm);
228  }
229  }
230  }
231  }
232  } else if (mTrapConfig->getTrapReg(TrapConfig::kFGBY) == 1) {
233  LOG(warn) << "you have asked to not use online gain calibrations but the selected trap config does have FGBY enabled. Gain calibrations will be 1, i.e. no effect";
234  }

267 bool digitindexcompare(unsigned int A, unsigned int B, const std::vector<o2::trd::Digit>& originalDigits)
268 {
269 
270  const o2::trd::Digit *a, *b;
271  a = &originalDigits[A];
272  b = &originalDigits[B];
273  FeeParam* fee = FeeParam::instance();
274  int rowa = a->getRow();
275  int rowb = b->getRow();
276  int pada = a->getPad();
277  int padb = b->getPad();
278  double timea = a->getTimeStamp();
279  double timeb = b->getTimeStamp();
280  int roba = fee->getROBfromPad(rowa, pada);
281  int robb = fee->getROBfromPad(rowb, padb);
282  int mcma = fee->getMCMfromPad(rowa, pada);
283  int mcmb = fee->getMCMfromPad(rowb, padb);
284  if (a->getTimeStamp() < b->getTimeStamp())
285  return 1;
286  if (a->getTimeStamp() > b->getTimeStamp())
287  return 0;
288  // timestamps are equal
289  if (a->getDetector() < b->getDetector())
290  return 1;
291  if (a->getDetector() > b->getDetector())
292  return 0;
293  //detectors are equal
294  if (roba < robb)
295  return 1;
296  if (roba > robb)
297  return 0;
298  //rob is equal
299  if (mcma < mcmb)
300  return 1;
301  else
302  return 0;
303 }

305 void TRDDPLTrapSimulatorTask::fixTriggerRecords(std::vector<o2::trd::TriggerRecord>& trigRecord)
306 {
307  // Trigger records are coming with an extra one at the end, and the first one blank and the last 2 having the same bunch crossing information.
308  // This is temporary.
309 
310  // take the nth records DataRange, and insert it into the (n-1)th DataRange
311  // thereby realigning the bunch crossings and dataranges.
312  // drop the final entry.
313 
314  //sanity check -- this is only true if the first range is 0 to 0
315  if (trigRecord[0].getFirstEntry() == 0 && trigRecord[0].getNumberOfObjects() == 0) {
316  for (int i = 0; i < trigRecord.size() - 1; i++) {
317  trigRecord[i].setDataRange(trigRecord[i + 1].getFirstEntry(), trigRecord[i + 1].getNumberOfObjects());
318  }
319  //now drop the final triggerrecord.
320  trigRecord.pop_back();
321  } else {
322  LOG(warn) << "TriggerRecord fix requested, but inital TriggerRecord is not 0,0";
323  }

326 void TRDDPLTrapSimulatorTask::run(o2::framework::ProcessingContext& pc)
327 {
328  LOG(info) << "TRD Trap Simulator Device running over incoming message";
329 
330  //#ifdef WITH_OPENMP
331  // int maxthreads = omp_get_max_threads();
332  // mNumThreads = std::min(maxthreads, 8);
333  // LOG(INFO) << "TRD: Trapping with " << mNumThreads << " threads ";
334  //#endif
335  //TODO need to change to gsl::span.
336 
337  // get inputs for the TrapSimulator
338  // the digits are going to be sorted, we therefore need a copy of the vector rather than an object created
339  // directly on the input data, the output vector however is created directly inside the message
340  // memory thus avoiding copy by snapshot
341  // TODO: rather than sorting the digits, an index can be sorted and the read-only input data accessed
342  // via the sorted index
343  auto inputDigits = pc.inputs().get<gsl::span<o2::trd::Digit>>("digitinput");
344  std::vector<o2::trd::Digit> msgDigits(inputDigits.begin(), inputDigits.end());
345  // auto digits pc.outputs().make<std::vector<o2::trd::Digit>>(Output{"TRD", "TRKDIGITS", 0, Lifetime::Timeframe}, msgDigits.begin(), msgDigits.end());
346  auto digitMCLabels = pc.inputs().get<o2::dataformats::MCTruthContainer<o2::MCCompLabel>*>("labelinput");
347 
348  // auto rawDataOut = pc.outputs().make<char>(Output{"TRD", "RAWDATA", 0, Lifetime::Timeframe}, 1000); //TODO number is just a place holder until we start using it.
349  o2::dataformats::MCTruthContainer<o2::MCCompLabel> trackletMCLabels;
350 
351  // the returned object is read-only as it refers directly to the underlying raw input data
352  // need to make a copy because the object might be changed in fixTriggerRecords
353  auto inputTriggerRecords = pc.inputs().get<gsl::span<o2::trd::TriggerRecord>>("triggerrecords");
354  std::vector<o2::trd::TriggerRecord> triggerRecords(inputTriggerRecords.begin(), inputTriggerRecords.end());
355  uint64_t currentTriggerRecord = 0;
356 
357  for (auto& trig : triggerRecords) {
358  LOG(debug) << "Trigger Record ; " << trig.getFirstEntry() << " --> " << trig.getNumberOfObjects();
359  }
360  // fix incoming trigger records if requested.
361  if (mFixTriggerRecords)
362  fixTriggerRecords(triggerRecords);
363 
364  std::vector<o2::trd::TriggerRecord> trackletTriggerRecords = triggerRecords; // copy over the whole thing but we only really want the bunch crossing info.
365 
366  //TODO these must be created directly in the output as done at the top of this run method
367  std::vector<unsigned int> msgDigitsIndex;
368  msgDigitsIndex.reserve(msgDigits.size());
369 
370  LOG(debug) << "Read in msgDigits with size of : " << msgDigits.size() << " labels contain : " << digitMCLabels->getNElements() << " with and index size of : " << digitMCLabels->getIndexedSize();
371 
372  if (digitMCLabels->getIndexedSize() != msgDigits.size()) {
373  LOG(debug) << "Read in msgDigits with size of : " << msgDigits.size() << " labels contain : " << digitMCLabels->getNElements() << " with and index size of : " << digitMCLabels->getIndexedSize();
374  LOG(warn) << "Digits and Labels coming into TrapSimulator are of differing sizes, labels will be jibberish. ";
375  }
376  //set up structures to hold the returning tracklets.
377  std::vector<Tracklet> trapTracklets; //vector to store the retrieved tracklets from an trapsim object
378  std::vector<Tracklet> trapTrackletsAccum;
379  trapTracklets.reserve(30);
380  trapTrackletsAccum.reserve(msgDigits.size() / 3);
381  msgDigitsIndex.reserve(msgDigits.size());
382 
383  int count = 0;
384  //make msgDigitsIndex a simple vector of ascending numbers mapping trivially into the msgDigits vector.
385  for (int i = 0; i < msgDigits.size(); i++) {
386  msgDigitsIndex.push_back(i);
387  }
388  LOG(debug) << "msgdigitsindex is " << msgDigitsIndex.size();
389 
390  auto sortstart = std::chrono::high_resolution_clock::now();
391  //sort the digits array
392  std::stable_sort(msgDigits.begin(), msgDigits.end(), msgDigitSortComparator);
393  mSortingTime = std::chrono::high_resolution_clock::now() - sortstart;
394  LOG(info) << "TRD Digit Sorting took " << mSortingTime.count();
395  // sort from triggerRecords.getFirstEntry() to triggerRecords.getFirstEntry()+triggerRecords.getNumberOfObjects();
396  //check the incoming triggerrecords:
397  for (auto& trig : triggerRecords) {
398  LOG(debug) << "Trigger Record ; " << trig.getFirstEntry() << " --> " << trig.getNumberOfObjects();
399  }
400  for (auto& trig : trackletTriggerRecords) {
401  LOG(debug) << "Trigger Tracklet Record ; " << trig.getFirstEntry() << " --> " << trig.getNumberOfObjects();
402  }
403  //accounting variables for various things.
404  int olddetector = -1;
405  int oldrow = -1;
406  int oldpad = -1;
407  int loopindex = 0;
408  int counttrackletadditions = 0;
409  int oldsize = 0;
410  double trackletrate;
411  unsigned long oldtrackletcount = 0;
412 
413  // now to loop over the incoming digits.
414  auto digitloopstart = std::chrono::high_resolution_clock::now();
415  uint64_t digitcounter = 0;
416  for (auto digititerator = msgDigits.begin(); digititerator != msgDigits.end() /* && loopindex<300*/; ++digititerator) {
417  //in here we have an entire padrow which corresponds to 8 TRAPs.
418  //while on a single padrow, populate data structures in the 8 trapsimulator.
419  //on change of padrow
420  // fireup trapsim, do its thing with each 18 sequence of pads data that already exists inside the class from previous iterations of the loop
421  double digittime = digititerator->getTimeStamp();
422  int pad = digititerator->getPad();
423  int row = digititerator->getRow();
424  int detector = digititerator->getDetector();
425  int rob = mFeeParam->getROBfromPad(row, pad);
426  int mcm = mFeeParam->getMCMfromPad(row, pad);
427  LOG(debug3) << "calculated rob and mcm at top of loop with detector:row:pad:rob:mcm" << detector << ":" << row << ":" << pad << ":" << rob << ":" << mcm;
428  if (digititerator == msgDigits.begin()) { // first time in loop
429  oldrow = row;
430  olddetector = detector;
431  }
432 
433  //figure out which trigger record from digits we are on
434  if (digitcounter >= triggerRecords[currentTriggerRecord].getFirstEntry() + triggerRecords[currentTriggerRecord].getNumberOfObjects()) {
435  //trigger record changed.
436  // so increment the tracklet trigger records and fill accordingly for the now completed prior triggerrecord.
437  uint64_t triggerrecordstart = 0;
438  if (currentTriggerRecord == 0) { // for not the first one we can simply look back to the previous one to get the start.
439  triggerrecordstart = 0;
440  trackletTriggerRecords[currentTriggerRecord].setDataRange(triggerrecordstart, trapTrackletsAccum.size());
441  } else {
442  triggerrecordstart = trackletTriggerRecords[currentTriggerRecord - 1].getFirstEntry() + trackletTriggerRecords[currentTriggerRecord - 1].getNumberOfObjects();
443  trackletTriggerRecords[currentTriggerRecord].setDataRange(triggerrecordstart, trapTrackletsAccum.size() - triggerrecordstart);
444  }
445  currentTriggerRecord++; //move to next trigger record.
446  }
447 
448  if (olddetector != detector || oldrow != row) {
449  // we have gone over the pad row. //TODO ??? do we need to check for change of time as well?
450  //all data is inside the 8 relavent trapsimulators
451  int preivousrob = mFeeParam->getROBfromPad(oldrow, oldpad); //
452  //fireup Trapsim.
453  auto traploopstart = std::chrono::high_resolution_clock::now();
454  unsigned long numberofusedtraps = 0;
455  for (int trapcounter = 0; trapcounter < 8; trapcounter++) {
456  unsigned int isinit = mTrapSimulator[trapcounter].checkInitialized();
457  // LOG(debug3) << "Start of trap : " << trapcounter;
458  if (mTrapSimulator[trapcounter].isDataSet()) { //firedtraps
459  //this one has been filled with data for the now previous pad row.
460  auto trapsimtimerstart = std::chrono::high_resolution_clock::now();
461  mTrapUsedCounter[trapcounter]++;
462  numberofusedtraps++;
463  mTrapSimulator[trapcounter].filter();
464  mTrapSimulator[trapcounter].tracklet();
465 
466  trapTracklets = mTrapSimulator[trapcounter].getTrackletArray(); //TODO remove the copy and send the Accumulated array into the Trapsimulator
467  auto trapLabels = mTrapSimulator[trapcounter].getTrackletLabels();
468 
469  LOG(debug) << "Tracklets accumulated before addition of new ones :" << trapTrackletsAccum.size()
470  << " :: about to add " << trapTracklets.size() << " count tracklets, labels coming in index of: "
471  << trapLabels.getIndexedSize() << " and elements of : "
472  << trapLabels.getNElements() << " with countrackletadditions : " << counttrackletadditions;
473  counttrackletadditions++;
