55 // Creates the sink for FairLogger / InfoLogger integration
56 auto createInfoLoggerSinkHelper(InfoLogger* logger, InfoLoggerContext* ctx)
57 {
58  return [logger,
59  ctx](const std::string& content, const fair::LogMetaData& metadata) {
60  // translate FMQ metadata
61  InfoLogger::InfoLogger::Severity severity = InfoLogger::Severity::Undefined;
62  int level = InfoLogger::undefinedMessageOption.level;
63 
64  if (metadata.severity_name == fair::Logger::SeverityName(fair::Severity::nolog)) {
65  // discard
66  return;
67  } else if (metadata.severity_name == fair::Logger::SeverityName(fair::Severity::fatal)) {
68  severity = InfoLogger::Severity::Fatal;
69  } else if (metadata.severity_name == fair::Logger::SeverityName(fair::Severity::error)) {
70  severity = InfoLogger::Severity::Error;
71  } else if (metadata.severity_name == fair::Logger::SeverityName(fair::Severity::warn)) {
72  severity = InfoLogger::Severity::Warning;
73  } else if (metadata.severity_name == fair::Logger::SeverityName(fair::Severity::state)) {
74  severity = InfoLogger::Severity::Info;
75  level = 10;
76  } else if (metadata.severity_name == fair::Logger::SeverityName(fair::Severity::info)) {
77  severity = InfoLogger::Severity::Info;
78  } else if (metadata.severity_name == fair::Logger::SeverityName(fair::Severity::debug)) {
79  severity = InfoLogger::Severity::Debug;
80  } else if (metadata.severity_name == fair::Logger::SeverityName(fair::Severity::debug1)) {
81  severity = InfoLogger::Severity::Debug;
82  level = 10;
83  } else if (metadata.severity_name == fair::Logger::SeverityName(fair::Severity::debug2)) {
84  severity = InfoLogger::Severity::Debug;
85  level = 20;
86  } else if (metadata.severity_name == fair::Logger::SeverityName(fair::Severity::debug3)) {
87  severity = InfoLogger::Severity::Debug;
88  level = 30;
89  } else if (metadata.severity_name == fair::Logger::SeverityName(fair::Severity::debug4)) {
90  severity = InfoLogger::Severity::Debug;
91  level = 40;
92  } else if (metadata.severity_name == fair::Logger::SeverityName(fair::Severity::trace)) {
93  severity = InfoLogger::Severity::Debug;
94  level = 50;
95  }
96 
97  InfoLogger::InfoLoggerMessageOption opt = {
98  severity,
99  level,
100  InfoLogger::undefinedMessageOption.errorCode,
101  metadata.file.c_str(),
102  atoi(metadata.line.c_str())};
103 
104  if (logger) {
105  logger->log(opt, *ctx, "DPL: %s", content.c_str());
106  }
107  };
108 };

110 o2::framework::ServiceSpec CommonServices::infologgerSpec()
111 {
112  return ServiceSpec{"infologger",
113  [](ServiceRegistry& services, DeviceState&, fair::mq::ProgOptions& options) -> ServiceHandle {
114  auto infoLoggerMode = options.GetPropertyAsString("infologger-mode");
115  if (infoLoggerMode != "") {
116  setenv("INFOLOGGER_MODE", infoLoggerMode.c_str(), 1);
117  }
118  auto infoLoggerService = new InfoLogger;
119  auto infoLoggerContext = &services.get<InfoLoggerContext>();
120 
121  auto infoLoggerSeverity = options.GetPropertyAsString("infologger-severity");
122  if (infoLoggerSeverity != "") {
123  fair::Logger::AddCustomSink("infologger", infoLoggerSeverity, createInfoLoggerSinkHelper(infoLoggerService, infoLoggerContext));
124  }
125  return ServiceHandle{TypeIdHelpers::uniqueId<InfoLogger>(), infoLoggerService};
126  },
127  noConfiguration(),
128  ServiceKind::Serial};
129 }

140  return ServiceHandle{TypeIdHelpers::uniqueId<ConfigurationInterface>(),
