51  void registerService(hash_type typeHash, void* service)
52  {
53  hash_type id = typeHash & MAX_SERVICES_MASK;
54  for (uint8_t i = 0; i < MAX_DISTANCE; ++i) {
55  if (mServicesValue[i + id] == nullptr) {
56  mServicesKey[i + id] = typeHash;
57  mServicesValue[i + id] = service;
58  return;
59  }
60  }
61  O2_BUILTIN_UNREACHABLE();
62  }

64  // Register a service for the given interface T
65  // with actual implementation C, i.e. C is derived from T.
66  // Only one instance of type C can be registered per type T.
67  // The fact we use a bare pointer indicates that the ownership
68  // of the service still belongs to whatever created it, and is
69  // not passed to the registry. It's therefore responsibility of
70  // the creator of the service to properly dispose it.
71  template <class I, class C>
72  void registerService(C* service)
73  {
74  // This only works for concrete implementations of the type T.
75  // We need type elision as we do not want to know all the services in
76  // advance
77  static_assert(std::is_base_of<I, C>::value == true,
78  "Registered service is not derived from declared interface");
79  constexpr hash_type typeHash = TypeIdHelpers::uniqueId<I>();
80  registerService(typeHash, reinterpret_cast<void*>(service));
81  }
82 
83  template <class I, class C>
84  void registerService(C const* service)
85  {
86  // This only works for concrete implementations of the type T.
87  // We need type elision as we do not want to know all the services in
88  // advance
89  static_assert(std::is_base_of<I, C>::value == true,
90  "Registered service is not derived from declared interface");
91  constexpr auto typeHash = TypeIdHelpers::uniqueId<I const>();
92  constexpr auto id = typeHash & MAX_SERVICES_MASK;
93  for (uint8_t i = 0; i < MAX_DISTANCE; ++i) {
94  if (mServicesValue[i + id] == nullptr) {
95  mServicesKey[i + id] = typeHash;
96  mServicesValue[i + id] = (void*)(service);
97  return;
98  }
99  }
100  O2_BUILTIN_UNREACHABLE();
101  }

104  template <typename T>
105  std::enable_if_t<std::is_const_v<T> == false, bool> active() const
106  {
107  constexpr auto typeHash = TypeIdHelpers::uniqueId<T>();
108  auto ptr = get(typeHash);
109  return ptr != nullptr;
110  }

115  template <typename T>
116  T& get() const
117  {
118  constexpr auto typeHash = TypeIdHelpers::uniqueId<T>();
119  auto ptr = get(typeHash);
120  if (O2_BUILTIN_LIKELY(ptr != nullptr)) {
121  if constexpr (std::is_const_v<T>) {
122  return *reinterpret_cast<T const*>(ptr);
123  } else {
124  return *reinterpret_cast<T*>(ptr);
125  }
126  }
127  throw std::runtime_error(std::string("Unable to find service of kind ") +
128  typeid(T).name() +
129  ". Make sure you use const / non-const correctly.");
130  }
131 
132  void* get(uint32_t typeHash) const
133  {
134  auto id = typeHash & MAX_SERVICES_MASK;
135  for (uint8_t i = 0; i < MAX_DISTANCE; ++i) {
136  if (mServicesKey[i + id] == typeHash) {
137  return mServicesValue[i + id];
138  }
139  }
140  return nullptr;
141  }
142 
143  private:
144  std::array<uint32_t, MAX_SERVICES + MAX_DISTANCE> mServicesKey;
145  std::array<void*, MAX_SERVICES + MAX_DISTANCE> mServicesValue;
146 };
