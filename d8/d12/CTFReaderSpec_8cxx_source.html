38 bool readFromTree(TTree& tree, const std::string brname, T& dest, int ev = 0)
39 {
40  auto* br = tree.GetBranch(brname.c_str());
41  if (br && br->GetEntries() > ev) {
42  auto* ptr = &dest;
43  br->SetAddress(&ptr);
44  br->GetEntry(ev);
45  br->ResetAddress();
46  return true;
47  }
48  return false;
49 }

65 void CTFReaderSpec::run(ProcessingContext& pc)
66 {
67  if (mNextToProcess >= mInput.size()) {
68  return;
69  }
70 
71  auto cput = mTimer.CpuTime();
72  mTimer.Start(false);
73  const auto& inputFile = mInput[mNextToProcess];
74  LOG(INFO) << "Reading CTF input " << mNextToProcess << ' ' << inputFile;
75 
76  TFile flIn(inputFile.c_str());
77  if (!flIn.IsOpen() || flIn.IsZombie()) {
78  LOG(ERROR) << "Failed to open file " << inputFile;
79  throw std::runtime_error("failed to open CTF file");
80  }
81  std::unique_ptr<TTree> tree((TTree*)flIn.Get(std::string(o2::base::NameConf::CTFTREENAME).c_str()));
82  if (!tree) {
83  throw std::runtime_error("failed to load CTF tree");
84  }
85  CTFHeader ctfHeader;
86  if (!readFromTree(*tree, "CTFHeader", ctfHeader)) {
87  throw std::runtime_error("did not find CTFHeader");
88  }
89  LOG(INFO) << ctfHeader;
90 
91  auto setFirstTFOrbit = [&](const std::string& label) {
92  auto* hd = pc.outputs().findMessageHeader({label});
93  if (!hd) {
94  throw std::runtime_error(o2::utils::concat_string("failed to find output message header for ", label));
95  }
96  hd->firstTForbit = ctfHeader.firstTForbit;
97  };
98 
99  // send CTF Header
100  pc.outputs().snapshot({"header"}, ctfHeader);
101  setFirstTFOrbit("header");
102 
103  DetID::mask_t detsTF = mDets & ctfHeader.detectors;
104  DetID det;
105 
106  det = DetID::ITS;
107  if (detsTF[det]) {
108  auto& bufVec = pc.outputs().make<std::vector<o2::ctf::BufferType>>({det.getName()}, sizeof(o2::itsmft::CTF));
109  o2::itsmft::CTF::readFromTree(bufVec, *(tree.get()), det.getName());
110  setFirstTFOrbit(det.getName());
111  }
112 
113  det = DetID::MFT;
114  if (detsTF[det]) {
115  auto& bufVec = pc.outputs().make<std::vector<o2::ctf::BufferType>>({det.getName()}, sizeof(o2::itsmft::CTF));
116  o2::itsmft::CTF::readFromTree(bufVec, *(tree.get()), det.getName());
117  setFirstTFOrbit(det.getName());
118  }
119 
120  det = DetID::TPC;
121  if (detsTF[det]) {
122  auto& bufVec = pc.outputs().make<std::vector<o2::ctf::BufferType>>({det.getName()}, sizeof(o2::tpc::CTF));
123  o2::tpc::CTF::readFromTree(bufVec, *(tree.get()), det.getName());
124  setFirstTFOrbit(det.getName());
125  }
126 
127  mTimer.Stop();
128  LOG(INFO) << "Read CTF " << inputFile << " in " << mTimer.CpuTime() - cput << " s";
129 
130  if (++mNextToProcess >= mInput.size()) {
131  pc.services().get<ControlService>().endOfStream();
132  pc.services().get<ControlService>().readyToQuit(QuitRequest::Me);
133  LOGF(INFO, "CTF reading total timing: Cpu: %.3e Real: %.3e s in %d slots",
134  mTimer.CpuTime(), mTimer.RealTime(), mTimer.Counter() - 1);
135  }
136 }

139 DataProcessorSpec getCTFReaderSpec(DetID::mask_t dets, const std::string& inp)
140 {
141  std::vector<OutputSpec> outputs;
142 
143  outputs.emplace_back(OutputLabel{"header"}, "CTF", "HEADER", 0, Lifetime::Timeframe);
144  for (auto id = DetID::First; id <= DetID::Last; id++) {
145  if (dets[id]) {
146  DetID det(id);
147  outputs.emplace_back(OutputLabel{det.getName()}, det.getDataOrigin(), "CTFDATA", 0, Lifetime::Timeframe);
148  }
149  }
150  return DataProcessorSpec{
151  "ctf-reader",
152  Inputs{},
153  outputs,
154  AlgorithmSpec{adaptFromTask<CTFReaderSpec>(dets, inp)},
155  Options{}};
156 }
