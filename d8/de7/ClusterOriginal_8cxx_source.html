142 std::pair<double, double> ClusterOriginal::minPadDimensions(int plane, int statusMask, bool matchMask) const
143 {
146 
147  assert(plane == 0 || plane == 1);
148 
149  double xmin(std::numeric_limits<float>::max());
150  double ymin(std::numeric_limits<float>::max());
151 
152  if (mMultiplicity[plane] == 0) {
153  return std::make_pair(xmin, ymin);
154  }
155 
156  for (const auto& pad : mPads) {
157  if (shouldUsePad(pad, plane, statusMask, matchMask)) {
158  xmin = TMath::Min(xmin, pad.dx());

167 void ClusterOriginal::area(int plane, double area[2][2]) const
168 {
171 
172  assert(plane == 0 || plane == 1);
173 
174  area[0][0] = std::numeric_limits<float>::max();
175  area[0][1] = -std::numeric_limits<float>::max();
176  area[1][0] = std::numeric_limits<float>::max();
177  area[1][1] = -std::numeric_limits<float>::max();
178 
179  for (const auto& pad : mPads) {
180  if (pad.plane() == plane) {
181  area[0][0] = TMath::Min(area[0][0], pad.x() - pad.dx());
182  area[0][1] = TMath::Max(area[0][1], pad.x() + pad.dx());
183  area[1][0] = TMath::Min(area[1][0], pad.y() - pad.dy());
184  area[1][1] = TMath::Max(area[1][1], pad.y() + pad.dy());
185  }
186  }

190 std::pair<int, int> ClusterOriginal::sizeInPads(int statusMask) const
191 {
194 
195  std::pair<double, double> dim0 = minPadDimensions(0, statusMask, true);
196  std::pair<double, double> dim1 = minPadDimensions(1, statusMask, true);
197 
198  std::pair<int, int> npad0 = sizeInPads(0, statusMask);
199  std::pair<int, int> npad1 = sizeInPads(1, statusMask);
200 
201  int nx(0);
202  if (TMath::Abs(dim0.first - dim1.first) < 1.e-3) {
203  nx = TMath::Max(npad0.first, npad1.first);
204  } else {
205  nx = dim0.first < dim1.first ? npad0.first : npad1.first;
206  }
207 
208  int ny(0);
209  if (TMath::Abs(dim0.second - dim1.second) < 1.e-3) {
210  ny = TMath::Max(npad0.second, npad1.second);
211  } else {
212  ny = dim0.second < dim1.second ? npad0.second : npad1.second;
213  }
214 
215  return std::make_pair(nx, ny);
216 }

219 std::pair<int, int> ClusterOriginal::sizeInPads(int plane, int statusMask) const
220 {
222 
223  assert(plane == 0 || plane == 1);
224 
225  if (mMultiplicity[plane] == 0) {
226  return std::make_pair(0, 0);
227  }
228 
229  // order pads in x and y directions considering positions closer than 0.01 cm as equal
230  auto cmp = [](double a, double b) { return a < b - 0.01; };
231  std::set<double, decltype(cmp)> padx(cmp);
232  std::set<double, decltype(cmp)> pady(cmp);
233 
234  for (const auto& pad : mPads) {
235  if (shouldUsePad(pad, plane, statusMask, true)) {
236  padx.emplace(pad.x());
237  pady.emplace(pad.y());
238  }
239  }
240 
241  return std::make_pair(padx.size(), pady.size());
242 }
