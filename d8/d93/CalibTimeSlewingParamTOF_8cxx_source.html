39 float CalibTimeSlewingParamTOF::evalTimeSlewing(int channel, float totIn) const
40 {
41 
42  // totIn is in ns
43  // the correction is returned in ps
44 
45  int sector = channel / NCHANNELXSECTOR;
46  channel = channel % NCHANNELXSECTOR;
47 
48  if (sector >= NSECTORS)
49  return 0.; // something went wrong!
50 
51  int n = mChannelStart[sector][channel];
52  if (n < 0)
53  return 0.;
54 
55  int nstop = (mTimeSlewing[sector]).size();
56  if (channel < NCHANNELXSECTOR - 1)
57  nstop = mChannelStart[sector][channel + 1];
58 
59  if (n >= nstop)
60  return 0.; // something went wrong!
61 
62  if (totIn == 0) {
63  return (float)((mTimeSlewing[sector])[n].second);
64  }
65 
66  // we convert tot from ns to ps and to unsigned short
67  unsigned short tot = (unsigned short)(totIn * 1000);
68 
69  while (n < nstop && tot > (mTimeSlewing[sector])[n].first) {
70  n++;
71  }
72  n--;
73 
74  if (n < 0) { // tot is lower than the first available value
75  return 0;
76  }
77 
78  if (n == nstop - 1)
79  return (float)((mTimeSlewing[sector])[n].second); // use the last value stored for that channel
80 
81  float w1 = (float)(tot - (mTimeSlewing[sector])[n].first);
82  float w2 = (float)((mTimeSlewing[sector])[n + 1].first - tot);
83 
84  return (float)(((mTimeSlewing[sector])[n].second * w2 + (mTimeSlewing[sector])[n + 1].second * w1) / ((mTimeSlewing[sector])[n + 1].first - (mTimeSlewing[sector])[n].first));
85 }

88 void CalibTimeSlewingParamTOF::addTimeSlewingInfo(int channel, float tot, float time)
89 {
90  // WE ARE ASSUMING THAT:
91  // channels have to be filled in increasing order (within the sector)
92  // tots have to be filled in increasing order (within the channel)
93 
94  // tot here is provided in ns, time in ps;
95  // tot will have to be converted into ps;
96  // type will have to be converted to unsigned short (tot) and short (time)
97 
98  int sector = channel / NCHANNELXSECTOR;
99  channel = channel % NCHANNELXSECTOR;
100 
101  // printf("DBG: addTimeSlewinginfo sec=%i\n",sector);
102 
103  if (sector >= NSECTORS)
104  return; // something went wrong!
105 
106  int currentch = channel;
107  while (mChannelStart[sector][currentch] == -1 && currentch > -1) {
108  // printf("DBG: fill channel %i\n",currentch);
109  // set also all the previous ones which were not filled
110  mChannelStart[sector][currentch] = (mTimeSlewing[sector]).size();
111  currentch--;
112  }
113  // printf("DBG: emplace back (%f,%f)\n",tot,time);
114  (mTimeSlewing[sector]).emplace_back((unsigned short)(tot * 1000), (short)time);
115 }

118 bool CalibTimeSlewingParamTOF::updateOffsetInfo(int channel, float residualOffset)
119 {
120 
121  // to update only the channel offset info in an existing CCDB object
122  // residual offset is given in ps
123 
124  int sector = channel / NCHANNELXSECTOR;
125  channel = channel % NCHANNELXSECTOR;
126  // printf("sector = %d, channel = %d\n", sector, channel);
127 
128  // printf("DBG: addTimeSlewinginfo sec=%i\n",sector);
129 
130  int n = mChannelStart[sector][channel]; // first time slewing entry for the current channel. this corresponds to tot = 0
131  if ((mTimeSlewing[sector])[n].first != 0) {
132  printf("DBG: there was no time offset set yet! first tot is %d\n", (mTimeSlewing[sector])[n].first);
133  std::pair<unsigned short, short> offsetToBeInserted(0, (short)residualOffset);
134  auto it = (mTimeSlewing[sector]).begin();
135  (mTimeSlewing[sector]).insert(it + n, offsetToBeInserted);
136  // now we have to increase by 1 all the mChannelStart for the channels that come after this
137  for (auto ch = channel + 1; ch < NCHANNELXSECTOR; ch++) {
138  mChannelStart[sector][ch]++;
139  }
140  return false;
141  }
142  (mTimeSlewing[sector])[n].second += (short)residualOffset;
143  return true;
