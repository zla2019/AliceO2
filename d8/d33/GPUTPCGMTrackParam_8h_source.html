274 GPUdi() float GPUTPCGMTrackParam::GetMirroredY(float Bz) const
275 {
276  // get Y of the point which has the same X, but located on the other side of trajectory
277  float qptBz = GetQPt() * Bz;
278  float cosPhi2 = 1.f - GetSinPhi() * GetSinPhi();
279  if (CAMath::Abs(qptBz) < 1.e-8f) {
280  qptBz = 1.e-8f;
281  }
282  if (cosPhi2 < 0.f) {
283  cosPhi2 = 0.f;
284  }
285  return GetY() - 2.f * sqrt(cosPhi2) / qptBz;
286 }
