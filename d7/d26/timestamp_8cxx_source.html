28 struct TimestampTask {
29  Produces<aod::Timestamps> ts_table;
30  RunToTimestamp* converter = nullptr;
31  Service<o2::ccdb::BasicCCDBManager> ccdb;
32  Configurable<std::string> path{"ccdb-path", "Test/RunToTimestamp", "path to the ccdb object"};
33  Configurable<long> timestamp{"ccdb-timestamp", -1, "timestamp of the object"};
34 
35  void init(o2::framework::InitContext&)
36  {
37  LOGF(info, "Initializing TimestampTask");
38  converter = ccdb->get<RunToTimestamp>(path.value);
39  if (converter) {
40  LOGF(info, "Run-number to timestamp converter found!");
41  } else {
42  LOGF(fatal, "Cannot find run-number to timestamp converter in path '%s'.", path.value.data());
43  }
44  }
45 
46  void process(aod::BC const& bc)
47  {
48  long timestamp = converter->getTimestamp(bc.runNumber());
49  InteractionRecord current(bc.globalBC(), 0);
50  InteractionRecord initial = o2::raw::HBFUtils::Instance().getFirstIR();
51  timestamp += 1000 * (current - initial).bc2ns();
52  ts_table(timestamp);
53  }
54 };
