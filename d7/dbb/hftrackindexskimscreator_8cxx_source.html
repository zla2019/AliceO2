42 struct SelectTracks {
43  Produces<aod::SelTrack> seltrack;
44  Configurable<double> ptmintrack{"ptmintrack", -1, "ptmin single track"};
45  Configurable<double> dcatoprimxymin{"dcatoprimxymin", 0, "dca xy to prim vtx min"};
46  Configurable<int> d_tpcnclsfound{"d_tpcnclsfound", 70, "min number of tpc cls >="};
47  Configurable<double> d_bz{"d_bz", 5.0, "bz field"};
48  Configurable<bool> b_dovalplots{"b_dovalplots", true, "do validation plots"};
49  OutputObj<TH1F> hpt_nocuts{TH1F("hpt_nocuts", "pt tracks (#GeV)", 100, 0., 10.)};
50  OutputObj<TH1F> hpt_cuts{TH1F("hpt_cuts", "pt tracks (#GeV)", 100, 0., 10.)};
51  OutputObj<TH1F> hdcatoprimxy_cuts{TH1F("hdcatoprimxy_cuts", "dca xy to prim. vertex (cm)", 100, -1.0, 1.0)};
52 
53  void process(aod::Collision const& collision,
54  soa::Join<aod::Tracks, aod::TracksCov, aod::TracksExtra> const& tracks)
55  {
56  Point3D<float> vtxXYZ(collision.posX(), collision.posY(), collision.posZ());
57  for (auto it0 = tracks.begin(); it0 != tracks.end(); ++it0) {
58  auto& track_0 = *it0;
59  int status = 1;
60  if (b_dovalplots == true)
61  hpt_nocuts->Fill(track_0.pt());
62  if (track_0.pt() < ptmintrack)
63  status = 0;
64  UChar_t clustermap_0 = track_0.itsClusterMap();
65  bool isselected_0 = track_0.tpcNClsFound() >= d_tpcnclsfound && track_0.flags() & 0x4;
66  isselected_0 = isselected_0 && (TESTBIT(clustermap_0, 0) || TESTBIT(clustermap_0, 1));
67  if (!isselected_0)
68  status = 0;
69  array<float, 2> dca;
70  float x0_ = track_0.x();
71  float alpha0_ = track_0.alpha();
72  std::array<float, 5> arraypar0 = {track_0.y(), track_0.z(), track_0.snp(),
73  track_0.tgl(), track_0.signed1Pt()};
74  std::array<float, 15> covpar0 = {track_0.cYY(), track_0.cZY(), track_0.cZZ(),
75  track_0.cSnpY(), track_0.cSnpZ(),
76  track_0.cSnpSnp(), track_0.cTglY(), track_0.cTglZ(),
77  track_0.cTglSnp(), track_0.cTglTgl(),
78  track_0.c1PtY(), track_0.c1PtZ(), track_0.c1PtSnp(),
79  track_0.c1PtTgl(), track_0.c1Pt21Pt2()};
80  o2::track::TrackParCov trackparvar0(x0_, alpha0_, arraypar0, covpar0);
81  trackparvar0.propagateParamToDCA(vtxXYZ, d_bz, &dca);
82  if (abs(dca[0]) < dcatoprimxymin)
83  status = 0;
84  if (b_dovalplots == true) {
85  if (status == 1) {
86  hpt_cuts->Fill(track_0.pt());
87  hdcatoprimxy_cuts->Fill(dca[0]);
88  }
89  }
90  seltrack(status, dca[0], dca[1]);
91  }
92  }
93 };
