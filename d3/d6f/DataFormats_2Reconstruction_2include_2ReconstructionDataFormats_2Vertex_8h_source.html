102 // Base primary vertex class, with position, error, N candidates and flags field
103 // The Stamp template parameter allows to define vertex (time)stamp in different

106 template <typename Stamp = o2::dataformats::TimeStamp<int>>
107 class Vertex : public VertexBase
108 {
109  public:
110  using ushort = unsigned short;
111  static ushort constexpr FlagsMask = 0xffff;
112 
113  Vertex() = default;
114  ~Vertex() = default;
115  Vertex(const Point3D<float>& pos, const std::array<float, kNCov>& cov, ushort nCont, float chi2)
116  : VertexBase(pos, cov), mNContributors(nCont), mChi2(chi2)
117  {
118  }
119 
120 #ifndef ALIGPU_GPUCODE
121  void print() const;
122 #endif
123 
124  ushort getNContributors() const { return mNContributors; }
125  void setNContributors(ushort v) { mNContributors = v; }
126 
127  ushort getBits() const { return mBits; }
128  bool isBitSet(int bit) const { return mBits & (FlagsMask & (0x1 << bit)); }
129  void setBits(ushort b) { mBits = b; }
130  void setBit(int bit) { mBits |= FlagsMask & (0x1 << bit); }
131  void resetBit(int bit) { mBits &= ~(FlagsMask & (0x1 << bit)); }
132 
133  void setChi2(float v) { mChi2 = v; }
134  float getChi2() const { return mChi2; }
135 
136  const Stamp& getTimeStamp() const { return mTimeStamp; }
137  Stamp& getTimeStamp() { return mTimeStamp; }
138  void setTimeStamp(const Stamp& v) { mTimeStamp = v; }
139 
140  private:
141  float mChi2 = 0;
142  ushort mNContributors = 0;
143  ushort mBits = 0;
144  Stamp mTimeStamp;
145 
146  ClassDefNV(Vertex, 2);
