37 void appendToTree(TTree& tree, const std::string brname, T& ptr)
38 {
39  auto* br = tree.GetBranch(brname.c_str());
40  auto* pptr = &ptr;
41  if (br) {
42  br->SetAddress(&pptr);
43  } else {
44  br = tree.Branch(brname.c_str(), &pptr);
45  }
46  br->Fill();
47  br->ResetAddress();
48 }

54 void CTFWriterSpec::run(ProcessingContext& pc)
55 {
56  auto cput = mTimer.CpuTime();
57  mTimer.Start(false);
58 
59  auto tfOrb = DataRefUtils::getHeader<o2::header::DataHeader*>(pc.inputs().getByPos(0))->firstTForbit;
60  TFile flOut(o2::base::NameConf::getCTFFileName(tfOrb).c_str(), "recreate");
61  TTree ctfTree(std::string(o2::base::NameConf::CTFTREENAME).c_str(), "O2 CTF tree");
62 
63  // create header
64  CTFHeader header{mRun, tfOrb};
65 
66  DetID det;
67 
68  det = DetID::ITS;
69  if (isPresent(det) && pc.inputs().isValid(det.getName())) {
70  auto ctfBuffer = pc.inputs().get<gsl::span<o2::ctf::BufferType>>(det.getName());
71  const auto ctfImage = o2::itsmft::CTF::getImage(ctfBuffer.data());
72  LOG(INFO) << "CTF for " << det.getName();
73  ctfImage.print();
74  ctfImage.appendToTree(ctfTree, det.getName());
75  header.detectors.set(det);
76  }
77 
78  det = DetID::MFT;
79  if (isPresent(det) && pc.inputs().isValid(det.getName())) {
80  auto ctfBuffer = pc.inputs().get<gsl::span<o2::ctf::BufferType>>(det.getName());
81  const auto ctfImage = o2::itsmft::CTF::getImage(ctfBuffer.data());
82  LOG(INFO) << "CTF for " << det.getName();
83  ctfImage.print();
84  ctfImage.appendToTree(ctfTree, det.getName());
85  header.detectors.set(det);
86  }
87 
88  det = DetID::TPC;
89  if (isPresent(det) && pc.inputs().isValid(det.getName())) {
90  auto ctfBuffer = pc.inputs().get<gsl::span<o2::ctf::BufferType>>(det.getName());
91  const auto ctfImage = o2::tpc::CTF::getImage(ctfBuffer.data());
92  LOG(INFO) << "CTF for " << det.getName();
93  ctfImage.print();
94  ctfImage.appendToTree(ctfTree, det.getName());
95  header.detectors.set(det);
96  }
97 
98  appendToTree(ctfTree, "CTFHeader", header);
99 
100  ctfTree.SetEntries(1);
101  ctfTree.Write();
102  flOut.Close();
103 
104  mTimer.Stop();
105  LOG(INFO) << "Wrote " << flOut.GetName() << " with CTF{" << header << "} in " << mTimer.CpuTime() - cput << " s";
106 }

114 DataProcessorSpec getCTFWriterSpec(DetID::mask_t dets, uint64_t run)
115 {
116  std::vector<InputSpec> inputs;
117  for (auto id = DetID::First; id <= DetID::Last; id++) {
118  if (dets[id]) {
119  inputs.emplace_back(DetID::getName(id), DetID::getDataOrigin(id), "CTFDATA", 0, Lifetime::Timeframe);
120  }
121  }
122  return DataProcessorSpec{
123  "ctf-writer",
124  inputs,
125  Outputs{},
126  AlgorithmSpec{adaptFromTask<CTFWriterSpec>(dets, run)},
127  Options{}};
128 }
