28 struct TimestampUserTask {
29  Service<o2::ccdb::BasicCCDBManager> ccdb;
30  Configurable<std::string> path{"ccdb-path", "qc/TOF/TOFTaskCompressed/hDiagnostic", "path to the ccdb object"};
31  Configurable<std::string> url{"ccdb-url", "http://ccdb-test.cern.ch:8080", "url of the ccdb repository"};
32  Configurable<long> nolaterthan{"ccdb-no-later-than", std::chrono::duration_cast<std::chrono::milliseconds>(std::chrono::system_clock::now().time_since_epoch()).count(), "latest acceptable timestamp of creation for the object"};
33 
34  void init(o2::framework::InitContext&)
35  {
36  // Set CCDB url
37  ccdb->setURL(url.value);
38  // Enabling object caching
39  ccdb->setCachingEnabled(true);
40  // Not later than now objects
41  ccdb->setCreatedNotAfter(nolaterthan.value);
42  }
43 
44  void process(soa::Join<aod::BCs, aod::Timestamps> const& iter)
45  {
46  for (auto i : iter) {
47  auto obj = ccdb->getForTimeStamp<TH2F>(path.value, i.timestamp());
48  if (obj) {
49  LOGF(info, "Found object!");
50  obj->Print("all");
51  } else {
52  LOGF(warning, "Object not found!");
53  }
54  }
55  }
56 };
