61 template <typename VEC>
62 void CTFCoder::encode(VEC& buff, const gsl::span<const ROFRecord>& rofRecVec, const gsl::span<const CompClusterExt>& cclusVec, const gsl::span<const unsigned char>& pattVec)
63 {
64  using MD = o2::ctf::Metadata::OptStore;
65  // what to do which each field: see o2::ctd::Metadata explanation
66  constexpr MD optField[CTF::getNBlocks()] = {
67  MD::EENCODE, //BLCfirstChipROF
68  MD::EENCODE, //BLCbcIncROF
69  MD::EENCODE, //BLCorbitIncROF
70  MD::EENCODE, //BLCnclusROF
71  MD::EENCODE, //BLCchipInc
72  MD::EENCODE, //BLCchipMul
73  MD::EENCODE, //BLCrow
74  MD::EENCODE, //BLCcolInc
75  MD::EENCODE, //BLCpattID
76  MD::EENCODE //BLCpattMap
77  };
78  CompressedClusters cc;
79  compress(cc, rofRecVec, cclusVec, pattVec);
80  auto ec = CTF::create(buff);
81  using ECB = CTF::base;
82 
83  ec->setHeader(cc.header);
84  ec->getANSHeader().majorVersion = 0;
85  ec->getANSHeader().minorVersion = 1;
86  // at every encoding the buffer might be autoexpanded, so we don't work with fixed pointer ec
87 #define ENCODE CTF::get(buff.data())->encode
88  // clang-format off
89  ENCODE(cc.firstChipROF, CTF::BLCfirstChipROF, o2::rans::ProbabilityBits16Bit, optField[CTF::BLCfirstChipROF], &buff);
90  ENCODE(cc.bcIncROF, CTF::BLCbcIncROF , o2::rans::ProbabilityBits16Bit, optField[CTF::BLCbcIncROF], &buff);
91  ENCODE(cc.orbitIncROF, CTF::BLCorbitIncROF, o2::rans::ProbabilityBits16Bit, optField[CTF::BLCorbitIncROF], &buff);
92  ENCODE(cc.nclusROF, CTF::BLCnclusROF, o2::rans::ProbabilityBits16Bit, optField[CTF::BLCnclusROF], &buff);
93  //
94  ENCODE(cc.chipInc, CTF::BLCchipInc, o2::rans::ProbabilityBits16Bit, optField[CTF::BLCchipInc], &buff);
95  ENCODE(cc.chipMul, CTF::BLCchipMul, o2::rans::ProbabilityBits16Bit, optField[CTF::BLCchipMul], &buff);
96  ENCODE(cc.row, CTF::BLCrow, o2::rans::ProbabilityBits16Bit, optField[CTF::BLCrow], &buff);
97  ENCODE(cc.colInc, CTF::BLCcolInc, o2::rans::ProbabilityBits16Bit, optField[CTF::BLCcolInc], &buff);
98  ENCODE(cc.pattID, CTF::BLCpattID, o2::rans::ProbabilityBits16Bit, optField[CTF::BLCpattID], &buff);
99  ENCODE(cc.pattMap, CTF::BLCpattMap, o2::rans::ProbabilityBits16Bit, optField[CTF::BLCpattMap], &buff);
100  // clang-format on
101 }

104 template <typename VROF, typename VCLUS, typename VPAT>
105 void CTFCoder::decode(const CTF::base& ec, VROF& rofRecVec, VCLUS& cclusVec, VPAT& pattVec)
106 {
107  CompressedClusters cc;
108  cc.header = ec.getHeader();
109  // clang-format off
110  ec.decode(cc.firstChipROF, CTF::BLCfirstChipROF);
111  ec.decode(cc.bcIncROF, CTF::BLCbcIncROF);
112  ec.decode(cc.orbitIncROF, CTF::BLCorbitIncROF);
113  ec.decode(cc.nclusROF, CTF::BLCnclusROF);
114  //
115  ec.decode(cc.chipInc, CTF::BLCchipInc);
116  ec.decode(cc.chipMul, CTF::BLCchipMul);
117  ec.decode(cc.row, CTF::BLCrow);
118  ec.decode(cc.colInc, CTF::BLCcolInc);
119  ec.decode(cc.pattID, CTF::BLCpattID);
120  ec.decode(cc.pattMap, CTF::BLCpattMap);
121  // clang-format on
122  //
123  decompress(cc, rofRecVec, cclusVec, pattVec);
124 }

127 template <typename VROF, typename VCLUS, typename VPAT>
128 void CTFCoder::decompress(const CompressedClusters& cc, VROF& rofRecVec, VCLUS& cclusVec, VPAT& pattVec)
129 {
130  rofRecVec.resize(cc.header.nROFs);
131  cclusVec.resize(cc.header.nClusters);
132  pattVec.resize(cc.header.nPatternBytes);
133 
134  o2::InteractionRecord prevIR(cc.header.firstBC, cc.header.firstOrbit);
135  uint32_t firstEntry = 0, clCount = 0, chipCount = 0;
136  for (uint32_t irof = 0; irof < cc.header.nROFs; irof++) {
137  // restore ROFRecord
138  auto& rofRec = rofRecVec[irof];
139  if (cc.orbitIncROF[irof]) { // new orbit
140  prevIR.bc = cc.bcIncROF[irof]; // bcInc has absolute meaning
141  prevIR.orbit += cc.orbitIncROF[irof];
142  } else {
143  prevIR.bc += cc.bcIncROF[irof];
144  }
145  rofRec.setBCData(prevIR);
146  rofRec.setFirstEntry(firstEntry);
147  rofRec.setNEntries(cc.nclusROF[irof]);
148  firstEntry += cc.nclusROF[irof];
149 
150  // resrore chips data
151  auto prevChip = cc.firstChipROF[irof];
152  uint16_t prevCol = 0, prevRow = 0;
153 
154  // >> this is the version with chipInc stored once per new chip
155  int inChip = 0;
156  for (uint32_t icl = 0; icl < cc.nclusROF[irof]; icl++) {
157  auto& clus = cclusVec[clCount];
158  if (inChip++ < cc.chipMul[chipCount]) { // still the same chip
159  clus.setCol((prevCol += cc.colInc[clCount]));
160  } else { // new chip starts
161  prevChip += cc.chipInc[++chipCount];
162  inChip = 1;
163  clus.setCol((prevCol = cc.colInc[clCount])); // colInc has abs. col meaning
164  }
165  clus.setChipID(prevChip);
166  clus.setRow(cc.row[clCount]);
167  clus.setPatternID(cc.pattID[clCount]);
168  clCount++;
169  }
170  if (cc.nclusROF[irof]) {
171  chipCount++; // since next chip for sure will be new and inChip will be 0...
172  }
173  // << this is the version with chipInc stored once per new chip
174 
175  /*
176  // >> this is the version with chipInc stored for every pixel, requires entropy compression dealing with repeating symbols
177  for (uint32_t icl = 0; icl < cc.nclusROF[irof]; icl++) {
178  auto& clus = cclusVec[clCount];
179  if (cc.chipInc[clCount]) { // new chip
180  prevChip += cc.chipInc[clCount];
181  clus.setCol((prevCol = cc.colInc[clCount])); // colInc has abs. col meaning
182  } else {
183  clus.setCol((prevCol += cc.colInc[clCount]));
184  }
185  clus.setChipID(prevChip);
186  clus.setRow(cc.row[clCount]);
187  clus.setPatternID(cc.pattID[clCount]);
188  clCount++;
189  }
190  // << this is the version with chipInc stored for every pixel, requires entropy compression dealing with repeating symbols
191  */
192  }
193  // explicit patterns
194  memcpy(pattVec.data(), cc.pattMap.data(), cc.header.nPatternBytes); // RSTODO use swap?
195  assert(chipCount == cc.header.nChips);
196 
197  if (clCount != cc.header.nClusters) {
198  LOG(ERROR) << "expected " << cc.header.nClusters << " but counted " << clCount << " in ROFRecords";
199  throw std::runtime_error("mismatch between expected and counter number of clusters");
200  }
201 }
