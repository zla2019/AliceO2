20 bool isStable(int pdg)
21 {
22  // Decide whether particle (pdg) is stable
23 
24  // All ions/nucleons are considered as stable
25  // Nuclear code is 10LZZZAAAI
26  if (pdg > 1000000000) {
27  return true;
28  }
29 
30  constexpr int kNstable = 18;
31  int pdgStable[kNstable] = {
32  kGamma, // Photon
33  kElectron, // Electron
34  kMuonPlus, // Muon
35  kPiPlus, // Pion
36  kKPlus, // Kaon
37  kK0Short, // K0s
38  kK0Long, // K0l
39  kProton, // Proton
40  kNeutron, // Neutron
41  kLambda0, // Lambda_0
42  kSigmaMinus, // Sigma Minus
43  kSigmaPlus, // Sigma Plus
44  3312, // Xsi Minus
45  3322, // Xsi
46  3334, // Omega
47  kNuE, // Electron Neutrino
48  kNuMu, // Muon Neutrino
49  kNuTau // Tau Neutrino
50  };
51 
52  for (int i = 0; i < kNstable; i++) {
53  if (pdg == std::abs(pdgStable[i])) {
54  return true;
55  }
56  }
57 
58  return false;
59 }

62 template <typename TMCParticle, typename TMCParticles>
63 bool isPhysicalPrimary(TMCParticles& mcParticles, TMCParticle const& particle)
64 {
65  // Test if a particle is a physical primary according to the following definition:
66  // Particles produced in the collision including products of strong and
67  // electromagnetic decay and excluding feed-down from weak decays of strange
68  // particles.
69 
70  LOGF(debug, "isPhysicalPrimary for %d", particle.index());
71 
72  const int ist = particle.statusCode();
73  const int pdg = std::abs(particle.pdgCode());
74 
75  // Initial state particle
76  // Solution for K0L decayed by Pythia6
77  // ->
78  if ((ist > 1) && (pdg != 130) && particle.producedByGenerator())
79  return false;
80  if ((ist > 1) && !particle.producedByGenerator())
81  return false;
82  // <-
83 
84  if (!isStable(pdg))
85  return false;
86 
87  if (particle.producedByGenerator()) {
88  // Solution for K0L decayed by Pythia6
89  // ->
90  if (particle.mother()[0] != -1) {
91  auto mother = mcParticles.iteratorAt(particle.mother()[0]);
92  if (std::abs(mother.pdgCode()) == 130)
93  return false;
94  }
95  // <-
96  // check for direct photon in parton shower
97  // ->
98  if (pdg == 22 && particle.daughter()[0] != -1) {
99  LOGF(debug, "D %d", particle.daughter()[0]);
100  auto daughter = mcParticles.iteratorAt(particle.daughter()[0]);
101  if (daughter.pdgCode() == 22)
102  return false;
103  }
104  // <-
105  return true;
106  }
107 
108  // Particle produced during transport
109 
110  LOGF(debug, "M0 %d %d", particle.producedByGenerator(), particle.mother()[0]);
111  auto mother = mcParticles.iteratorAt(particle.mother()[0]);
112  int mpdg = std::abs(mother.pdgCode());
113 
114  // Check for Sigma0
115  if ((mpdg == 3212) && mother.producedByGenerator())
116  return true;
117 
118  // Check if it comes from a pi0 decay
119  if ((mpdg == kPi0) && mother.producedByGenerator())
120  return true;
121 
122  // Check if this is a heavy flavor decay product
123  int mfl = int(mpdg / std::pow(10, int(std::log10(mpdg))));
124 
125  // Light hadron
126  if (mfl < 4)
127  return false;
128 
129  // Heavy flavor hadron produced by generator
130  if (mother.producedByGenerator())
131  return true;
132 
133  // To be sure that heavy flavor has not been produced in a secondary interaction
134  // Loop back to the generated mother
135  LOGF(debug, "M0 %d %d", mother.producedByGenerator(), mother.mother()[0]);
136  while (mother.mother()[0] != -1 && !mother.producedByGenerator()) {
137  mother = mcParticles.iteratorAt(mother.mother()[0]);
138  LOGF(debug, "M+ %d %d", mother.producedByGenerator(), mother.mother()[0]);
139  mpdg = std::abs(mother.pdgCode());
140  mfl = int(mpdg / std::pow(10, int(std::log10(mpdg))));
141  }
142 
143  if (mfl < 4)
144  return false;
145  else
146  return true;
147 }
