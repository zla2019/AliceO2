78  void* get(uint32_t typeHash) const
79  {
80  auto id = typeHash & MAX_CONTEXT_MASK;
81  for (uint8_t i = 0; i < MAX_DISTANCE; ++i) {
82  if (mRegistryKey[i + id] == typeHash) {
83  return mRegistryValue[i + id];
84  }
85  }
86  return nullptr;
87  }

89  void set(void* instance, uint32_t typeHash)
90  {
91  auto id = typeHash & MAX_CONTEXT_MASK;
92  for (uint8_t i = 0; i < MAX_DISTANCE; ++i) {
93  if (mRegistryValue[i + id] == nullptr) {
94  mRegistryKey[i + id] = typeHash;
95  mRegistryValue[i + id] = instance;
96  return;
97  }
98  }
99  }
