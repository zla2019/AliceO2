196 CoarseLocation getDetElemCoarseLocation(int detElemId)
197 {
198  auto t = o2::mch::getTransformation(detElemId, *gGeoManager);
199  Point3D<double> localTestPos{0.0, 0.0, 0.0}; // slat center
200 
201  if (detElemId < 500) {
202  // in the rough ballpark of the center
203  // of the quadrants
204  localTestPos.SetXYZ(60, 60, 0);
205  }
206 
207  // for slats around the middle (y closest to 0) we have to be a bit
208  // more precise, so take a given pad reference, chosen to be
209  // the most top right or most top left pad
210 
211  switch (detElemId) {
212  case 500:
213  case 509:
214  localTestPos.SetXYZ(-72.50, 19.75, 0.0); // ds 107
215  break;
216  case 600:
217  case 609:
218  localTestPos.SetXYZ(-77.50, 19.75, 0.0); // ds 108
219  break;
220  case 700:
221  case 713:
222  case 800:
223  case 813:
224  case 900:
225  case 913:
226  case 1000:
227  case 1013:
228  localTestPos.SetXYZ(95.0, -19.75, 0); // ds 104
229  break;
230  }
231  Point3D<double> master;
232 
233  t.LocalToMaster(localTestPos, master);
234  bool right = master.x() > 10.;
235  bool top = master.y() > -10.;
236 
237  return CoarseLocation{right, top};
238 }
