728 inline void GBTFrame::updateSyncCheck(SyncArray& syncArray)
729 {
730  const auto offset = mPrevHWpos ^ 4;
731 
732  for (int s = 0; s < 5; s++)
733  for (int h = 0; h < 4; h++) {
734  const auto hPos = h + offset; // set position of last filled HW
735  // shift in a 1 if the halfword is 0x15
736  if (mFrameHalfWords[s][hPos] == 0x15)
737  mSyncCheckRegister[s] = (mSyncCheckRegister[s] << 1) | 1;
738  // shift in a 0 if the halfword is 0xA
739  else if (mFrameHalfWords[s][hPos] == 0xA)
740  mSyncCheckRegister[s] = (mSyncCheckRegister[s] << 1);
741  // otherwise reset the register to 0
742  else
743  mSyncCheckRegister[s] = 0;
744  // std::cout << " SReg : " << s << " : " << std::hex << mSyncCheckRegister[s] << std::endl;
745  // check the register content for the SYNC pattern
746  // Patterh is : 1100.1100.1100.1100.1111.0000.1111.0000 = 0xCCCCF0F0
747  // alternative Pattern for first corrupted SYNC word : 0100.1100.1100.1100.1111.0000.1111.0000 = 4CCC.F0F0
748  if (mSyncCheckRegister[s] == 0xCCCCF0F0 or mSyncCheckRegister[s] == 0x4CCCF0F0 or mSyncCheckRegister[s] == 0x0CCCF0F0) {
749  syncArray[s].setPos(mPacketNum, mFrameNum, mFilePos, h);
750  };
751  };
752 }

787 inline void GBTFrame::getFrameHalfWords()
788 {
789  constexpr adc_t P[5][4] = {{19, 18, 17, 16}, {39, 38, 37, 36}, {63, 62, 61, 60}, {83, 82, 81, 80}, {107, 106, 105, 104}};
790  // i = Stream, j = Halfword
791  for (int i = 0; i < 5; i++) {
792  for (int j = 0; j < 4; j++) {
793  mFrameHalfWords[i][j + mPrevHWpos] = bit(P[i][j], adc_t(4)) |
794  bit(adc_t(P[i][j] - 4), adc_t(3)) |
795  bit(adc_t(P[i][j] - 8), adc_t(2)) |
796  bit(adc_t(P[i][j] - 12), adc_t(1)) |
797  bit(adc_t(P[i][j] - 16), adc_t(0));
798  // std::cout << " S : " << i << " H : " << j << " : " << std::hex << res << std::endl;
799  };
800  }
801  mPrevHWpos ^= 4; // toggle position of previous HW position
802 }

805 inline void GBTFrame::getAdcValues(ADCRawData& rawData)
806 {
807  // loop over all the 5 data streams
808  for (int s = 0; s < 5; s++) {
809  if (!mSyncPos[s].synched()) {
810  continue;
811  }
812 
813  // assemble the 2 ADC words from 4 half-words. Which halfwords are
814  // used depends on the position of the SYNC pattern.
815  const uint32_t pos = mSyncPos[s].getHalfWordPosition();
816  const uint32_t v0 = (mFrameHalfWords[s][(pos + 2 + mPrevHWpos) & 7] << 5) | mFrameHalfWords[s][(pos + 1 + mPrevHWpos) & 7];
817  const uint32_t v1 = (mFrameHalfWords[s][(pos + 4 + mPrevHWpos) & 7] << 5) | mFrameHalfWords[s][(pos + 3 + mPrevHWpos) & 7];
818 
819  // add the two ADC values for the current processed stream
820  rawData.add(s, v0, v1);
821  };
