89 inline void InteractionSampler::generateCollisionTimes(std::vector<o2::InteractionTimeRecord>& dest)
90 {
91  // fill vector with interaction records
92  dest.clear();
93  for (int i = dest.capacity(); i--;) {

99 inline void InteractionSampler::nextCollidingBC(int n)
100 {
102  if ((mCurrBCIdx += n) >= (int)mInteractingBCs.size()) {
103  mIR.orbit += mCurrBCIdx / mInteractingBCs.size();
104  mCurrBCIdx %= mInteractingBCs.size();
105  }
106  mIR.bc = mInteractingBCs[mCurrBCIdx];
107 }
