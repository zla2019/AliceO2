113  void expandPreCluster(uint32_t ip, uint16_t row, int preClusIndex)
114  {
115  auto& firstIndex = preClusterHeads[preClusterIndices[preClusIndex]];
116  pixels.emplace_back(firstIndex, ip);
117  firstIndex = pixels.size() - 1;
118  curr[row] = preClusIndex;
119  }
120 
122  void addNewPrecluster(uint32_t ip, uint16_t row)
123  {
124  preClusterHeads.push_back(pixels.size());
125  // new head does not point yet (-1) on other pixels, store just the entry of the pixel in the ChipPixelData
126  pixels.emplace_back(-1, ip);
127  int lastIndex = preClusterIndices.size();
128  preClusterIndices.push_back(lastIndex);
129  curr[row] = lastIndex; // store index of the new precluster in the current column buffer
130  }
