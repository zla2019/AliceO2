42 void VarManager::ResetValues(int startValue, int endValue)
43 {
44  //
45  // reset all variables to an "innocent" value
46  // NOTE: here we use -9999.0 as a neutral value, but depending on situation, this may not be the case

48  fgValues[i] = -9999.;
