44 // Function to update the temporal cluster
45 void KrBoxClusterFinder::updateTempClusterFinal()
46 {
47  const float oneOverQtot = 1. / mTempCluster.totCharge;
48  mTempCluster.meanPad *= oneOverQtot;
49  mTempCluster.sigmaPad *= oneOverQtot;
50  mTempCluster.meanRow *= oneOverQtot;
51  mTempCluster.sigmaRow *= oneOverQtot;
52  mTempCluster.meanTime *= oneOverQtot;
53  mTempCluster.sigmaTime *= oneOverQtot;
54  mTempCluster.sigmaPad = std::sqrt(std::abs(mTempCluster.sigmaPad - mTempCluster.meanPad * mTempCluster.meanPad));
55  mTempCluster.sigmaRow = std::sqrt(std::abs(mTempCluster.sigmaRow - mTempCluster.meanRow * mTempCluster.meanRow));
56  mTempCluster.sigmaTime = std::sqrt(std::abs(mTempCluster.sigmaTime - mTempCluster.meanTime * mTempCluster.meanTime));
57 }

59 // Function to update the temporal cluster.
60 void KrBoxClusterFinder::updateTempCluster(float tempCharge, int tempPad, int tempRow, int tempTime)
61 {
62  if (tempCharge < mQThreshold) {
63  LOGP(warning, "Update cluster was called but current charge is below mQThreshold");
64  return;
65  }
66 
67  mTempCluster.size += 1;
68  mTempCluster.totCharge += tempCharge;
69 
70  mTempCluster.meanPad += tempPad * tempCharge;
71  mTempCluster.sigmaPad += tempPad * tempPad * tempCharge;
72 
73  mTempCluster.meanRow += tempRow * tempCharge;
74  mTempCluster.sigmaRow += tempRow * tempRow * tempCharge;
75 
76  mTempCluster.meanTime += tempTime * tempCharge;
77  mTempCluster.sigmaTime += tempTime * tempTime * tempCharge;
78 
79  if (tempCharge > mTempCluster.maxCharge) {
80  mTempCluster.maxCharge = tempCharge;
81  }
82 }

84 // This function finds and evaluates all clusters in a 3D mMapOfAllDigits generated by the
85 // mMapOfAllDigitsCreator function, this function also updates the cluster tree
86 std::vector<std::tuple<int, int, int>> KrBoxClusterFinder::findLocalMaxima()
87 {
88  std::vector<std::tuple<int, int, int>> localMaximaCoords;
89  // loop over whole mMapOfAllDigits the find clusers
90  for (int iTime = 0; iTime < MaxTimes; iTime++) {
91  for (int iRow = 0; iRow < MaxRows; iRow++) {
92  for (int iPad = 0; iPad < MaxPads; iPad++) {
93 
94  mTempCluster.reset();
95  const float qMax = mMapOfAllDigits[iTime][iRow][iPad];
96 
97  // cluster Maximum must at least be larger than Threshold
98  if (qMax <= mQThresholdMax) {
99  continue;
100  }
101 
102  // Acceptance condition: Require at least mMinNumberOfNeighbours neigbours
103  // with signal in any direction!
104  int noNeighbours = 0;
105  if ((iPad + 1 < MaxPads) && (mMapOfAllDigits[iTime][iRow][iPad + 1] > mQThreshold)) {
106  noNeighbours++;
107  }
108  if ((iPad - 1 >= 0) && (mMapOfAllDigits[iTime][iRow][iPad - 1] > mQThreshold)) {
109  noNeighbours++;
110  }
111  if ((iTime + 1 < MaxTimes) && (mMapOfAllDigits[iTime + 1][iRow][iPad] > mQThreshold)) {
112  noNeighbours++;
113  }
114  if ((iTime - 1 >= 0) && (mMapOfAllDigits[iTime - 1][iRow][iPad] > mQThreshold)) {
115  noNeighbours++;
116  }
117  if ((iRow + 1 < MaxRows) && (mMapOfAllDigits[iTime][iRow + 1][iPad] > mQThreshold)) {
118  noNeighbours++;
119  }
120  if ((iRow - 1 >= 0) && (mMapOfAllDigits[iTime][iRow - 1][iPad] > mQThreshold)) {
121  noNeighbours++;
122  }
123  if (noNeighbours < mMinNumberOfNeighbours) {
124  continue;
125  }
126 
127  // Check that this is a local maximum
128  // Note that the checking is done so that if 2 charges have the same
129  // qMax then only 1 cluster is generated
130  // (that is why there is BOTH > and >=)
131  // -> only the maximum with the smalest indices will be accepted
132  bool thisIsMax = true;
133 
134  for (int i = -mMaxClusterSizePad; (i <= mMaxClusterSizePad) && thisIsMax;
135  i++) {
136  for (int k = -mMaxClusterSizeRow; (k <= mMaxClusterSizeRow) && thisIsMax;
137  k++) {
138  for (int j = -mMaxClusterSizeTime;
139  (j <= mMaxClusterSizeTime) && thisIsMax; j++) {
140  if ((iPad + i < MaxPads) && (iTime + j < MaxTimes) &&
141  (iRow + k < MaxRows) && (iPad + i >= 0) && (iTime + j) >= 0 &&
142  (iRow + k) >= 0 &&
143  mMapOfAllDigits[iTime + j][iRow + k][iPad + i] > qMax) {
144  thisIsMax = false;
145  }
146  }
147  }
148  }
149 
150  if (!thisIsMax) {
151  continue;
152  } else {
153  localMaximaCoords.emplace_back(std::make_tuple(iPad, iRow, iTime));
154  }
155  }
156  }
157  }
158  return localMaximaCoords;
159 }

161 // Calculate the total charge as the sum over the region:
162 //
163 // o o o o o
164 // o i i i o
165 // o i C i o
166 // o i i i o
167 // o o o o o
168 //
169 // with qmax at the center C.
170 //
171 // The inner charge (i) we always add, but we only add the outer
172 // charge (o) if the neighboring inner bin (i) has a signal.
173 //
174 
175 // for loop over whole cluster, to determine if a charge should be added
176 // conditions are extrapolation of the 5x5 cluster case to arbitrary
177 // cluster sizes in 3 dimensions
178 KrCluster KrBoxClusterFinder::buildCluster(int clusterCenterPad, int clusterCenterRow, int clusterCenterTime)
179 {
180  mTempCluster = KrCluster();
181 
182  for (int iTime = -mMaxClusterSizeTime; iTime <= mMaxClusterSizeTime; iTime++) {
183  for (int iRow = -mMaxClusterSizeRow; iRow <= mMaxClusterSizeRow; iRow++) {
184  for (int iPad = -mMaxClusterSizePad; iPad <= mMaxClusterSizePad; iPad++) {
185  // First: Check if we might check outside of map:
186  if (clusterCenterTime + iTime < 0 || clusterCenterTime + iTime >= MaxTimes || clusterCenterPad + iPad < 0 || clusterCenterPad + iPad >= MaxPads || clusterCenterRow + iRow < 0 || clusterCenterRow + iRow >= MaxRows || mMapOfAllDigits.at(clusterCenterTime + iTime).at(clusterCenterRow + iRow).at(clusterCenterPad + iPad) <= mQThreshold) {
187  continue;
188  }
189  // If not, there are several cases which were explained (for 2D) in the header of the code.
190  // The first one is for the diagonal. So, the digit we are investigating here is on the diagonal:
191  else if (std::abs(iTime) == std::abs(iPad) && std::abs(iTime) == std::abs(iRow)) {
192  // Now we check, if the next inner digit has a signal above threshold:
193  if (mMapOfAllDigits[clusterCenterTime + iTime - signnum(iTime)][clusterCenterRow + iRow - signnum(iRow)][clusterCenterPad + iPad - signnum(iPad)] > mQThreshold) {
194  // If yes, the cluster gets updated with the digit on the diagonal.
195  updateTempCluster(mMapOfAllDigits[clusterCenterTime + iTime][clusterCenterRow + iRow][clusterCenterPad + iPad], clusterCenterPad + iPad, clusterCenterRow + iRow, clusterCenterTime + iTime);
196  }
197  }
198  // Basically, we go through every possible case in the next few if-else conditions:
199  else if (std::abs(iTime) == std::abs(iPad)) {
200  if (mMapOfAllDigits[clusterCenterTime + iTime - signnum(iTime)][clusterCenterRow + iRow][clusterCenterPad + iPad - signnum(iPad)] > mQThreshold) {
201  updateTempCluster(mMapOfAllDigits[clusterCenterTime + iTime][clusterCenterRow + iRow][clusterCenterPad + iPad], clusterCenterPad + iPad, clusterCenterRow + iRow, clusterCenterTime + iTime);
202  }
203  } else if (std::abs(iTime) == std::abs(iRow)) {
204  if (mMapOfAllDigits[clusterCenterTime + iTime - signnum(iTime)][clusterCenterRow + iRow - signnum(iRow)][clusterCenterPad + iPad] > mQThreshold) {
205  updateTempCluster(mMapOfAllDigits[clusterCenterTime + iTime][clusterCenterRow + iRow][clusterCenterPad + iPad], clusterCenterPad + iPad, clusterCenterRow + iRow, clusterCenterTime + iTime);
206  }
207  } else if (std::abs(iPad) == std::abs(iRow)) {
208  if (mMapOfAllDigits[clusterCenterTime + iTime][clusterCenterRow + iRow - signnum(iRow)][clusterCenterPad + iPad - signnum(iPad)] > mQThreshold) {
209  updateTempCluster(mMapOfAllDigits[clusterCenterTime + iTime][clusterCenterRow + iRow][clusterCenterPad + iPad], clusterCenterPad + iPad, clusterCenterRow + iRow, clusterCenterTime + iTime);
210  }
211  } else if (std::abs(iTime) > std::abs(iPad) && std::abs(iTime) > std::abs(iRow)) {
212  if (mMapOfAllDigits[clusterCenterTime + iTime - signnum(iTime)][clusterCenterRow + iRow][clusterCenterPad + iPad] > mQThreshold) {
213  updateTempCluster(mMapOfAllDigits[clusterCenterTime + iTime][clusterCenterRow + iRow][clusterCenterPad + iPad], clusterCenterPad + iPad, clusterCenterRow + iRow, clusterCenterTime + iTime);
214  }
215  } else if (std::abs(iTime) < std::abs(iPad) && std::abs(iPad) > std::abs(iRow)) {
216  if (mMapOfAllDigits[clusterCenterTime + iTime][clusterCenterRow + iRow][clusterCenterPad + iPad - signnum(iPad)] > mQThreshold) {
217  updateTempCluster(mMapOfAllDigits[clusterCenterTime + iTime][clusterCenterRow + iRow][clusterCenterPad + iPad], clusterCenterPad + iPad, clusterCenterRow + iRow, clusterCenterTime + iTime);
218  }
219  } else if (std::abs(iTime) < std::abs(iRow) && std::abs(iPad) < std::abs(iRow)) {
220  if (mMapOfAllDigits[clusterCenterTime + iTime][clusterCenterRow + iRow - signnum(iRow)][clusterCenterPad + iPad] > mQThreshold) {
221  updateTempCluster(mMapOfAllDigits[clusterCenterTime + iTime][clusterCenterRow + iRow][clusterCenterPad + iPad], clusterCenterPad + iPad, clusterCenterRow + iRow, clusterCenterTime + iTime);
222  }
223  }
224  }
225  }
226  }
227  // At the end, out mTempCluster should contain all digits that were assigned to the cluster.
228  // So before returning it, we update it one last time to calculate the correct means and sigmas.
229  updateTempClusterFinal();
230  return mTempCluster;
231 }
