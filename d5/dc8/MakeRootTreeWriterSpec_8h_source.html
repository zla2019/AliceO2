331  // worker instance
332  std::shared_ptr<WriterType> writer;
333  // keys for branch name options
334  std::vector<std::pair<std::string, std::string>> branchNameOptions;
335  // number of events to be processed

338  std::unordered_set<std::string> activeInputs;
339  // event counter

342  TerminationPolicy terminationPolicy = TerminationPolicy::Process;

350  auto processAttributes = std::make_shared<ProcessAttributes>();
351  processAttributes->writer = mWriter;
352  processAttributes->branchNameOptions = mBranchNameOptions;
353  processAttributes->terminationCondition = std::move(mTerminationCondition);
354  processAttributes->preprocessor = std::move(mPreprocessor);
355 
356  // set the list of active inputs, every input which is indecated as 'ready' will be removed from the list
357  // the process is ready if the list is empty
358  for (auto const& input : mInputs) {
359  processAttributes->activeInputs.emplace(input.binding);

363  // the init function is returned to the DPL in order to init the process
364  auto initFct = [processAttributes, TerminationPolicyMap = TerminationPolicyMap](InitContext& ic) {
365  auto& branchNameOptions = processAttributes->branchNameOptions;
366  auto filename = ic.options().get<std::string>("outfile");
367  auto treename = ic.options().get<std::string>("treename");
368  auto treetitle = ic.options().get<std::string>("treetitle");
369  processAttributes->nEvents = ic.options().get<int>("nevents");
370  if (processAttributes->nEvents > 0 && processAttributes->activeInputs.size() != processAttributes->nofBranches) {
371  LOG(WARNING) << "the n inputs serve in total m branches with n != m, this means that there will be data for\n"
372  << "different branches on the same input. Be aware that the --nevents option might lead to incomplete\n"
373  << "data in the output file as the number of processed input sets is counted";
374  }
375  try {
376  processAttributes->terminationPolicy = TerminationPolicyMap.at(ic.options().get<std::string>("terminate"));
377  } catch (std::out_of_range&) {
378  throw std::invalid_argument(std::string("invalid termination policy: ") + ic.options().get<std::string>("terminate"));
379  }
380  if (filename.empty() || treename.empty()) {
381  throw std::invalid_argument("output file name and tree name are mandatory options");
382  }
383  for (size_t branchIndex = 0; branchIndex < branchNameOptions.size(); branchIndex++) {
384  // pair of key (first) - value (second)
385  if (branchNameOptions[branchIndex].first.empty()) {
386  continue;
387  }
388  auto branchName = ic.options().get<std::string>(branchNameOptions[branchIndex].first.c_str());
389  processAttributes->writer->setBranchName(branchIndex, branchName.c_str());
390  }
391  processAttributes->writer->init(filename.c_str(), treename.c_str(), treetitle.c_str());
392 
393  // the callback to be set as hook at stop of processing for the framework
394  auto finishWriting = [processAttributes]() {
395  processAttributes->writer->close();
396  };
397  ic.services().get<CallbackService>().set(CallbackService::Id::Stop, finishWriting);
398 
399  auto processingFct = [processAttributes](ProcessingContext& pc) {
400  auto& writer = processAttributes->writer;
401  auto& terminationPolicy = processAttributes->terminationPolicy;
402  auto& terminationCondition = processAttributes->terminationCondition;
403  auto& preprocessor = processAttributes->preprocessor;
404  auto& activeInputs = processAttributes->activeInputs;
405  auto& counter = processAttributes->counter;
406  auto& nEvents = processAttributes->nEvents;
407  if (writer->isClosed()) {
408  return;
409  }
410 
411  // if the termination condition contains function of type CheckProcessing, this is checked
412  // before the processing, currently implemented logic:
413  // - processing is skipped if this is indicated by at least one input
414  // - if input is 'ready' it is removed from the list of active inputs
415  auto checkProcessing = [&terminationCondition, &activeInputs](auto const& inputs) {
416  bool doProcessing = true;
417  if (std::holds_alternative<TerminationCondition::CheckProcessing>(terminationCondition.check)) {
418  auto& check = std::get<TerminationCondition::CheckProcessing>(terminationCondition.check);
419  for (auto const& ref : inputs) {
420  auto iter = activeInputs.find(ref.spec->binding);
421  auto [action, ready] = check(ref);
422  if (action == TerminationCondition::Action::SkipProcessing) {
423  // this condition tells us not to process the data any further
424  doProcessing = false;
425  }
426  if (iter != activeInputs.end() && ready) {
427  // this input is ready, remove from active inputs
428  activeInputs.erase(iter);
429  }
430  }
431  }
432  return doProcessing;
433  };
434  // if the termination condition contains function of type CheckReady, this is checked
435  // after the processing. All inputs marked 'ready' are removed from the list of active inputs,
436  // process treated as 'ready' if no active inputs
437  auto checkReady = [&terminationCondition, &activeInputs](auto const& inputs) {
438  if (std::holds_alternative<TerminationCondition::CheckReady>(terminationCondition.check)) {
439  auto& check = std::get<TerminationCondition::CheckReady>(terminationCondition.check);
440  for (auto const& ref : inputs) {
441  auto iter = activeInputs.find(ref.spec->binding);
442  if (iter != activeInputs.end() && check(ref)) {
443  // this input is ready, remove from active inputs
444  activeInputs.erase(iter);
445  }
446  }
447  }
448  return activeInputs.size() == 0;
449  };
450 
451  if (preprocessor) {
452  preprocessor(pc);
453  }
454  if (checkProcessing(pc.inputs())) {
455  (*writer)(pc.inputs());
456  counter = counter + 1;
457  }
458 
459  if ((nEvents >= 0 && counter == nEvents) || checkReady(pc.inputs())) {
460  writer->close();
461  pc.services().get<ControlService>().readyToQuit(terminationPolicy == TerminationPolicy::Workflow ? QuitRequest::All : QuitRequest::Me);
462  }
463  };
464 
465  return processingFct;
466  };

588  template <size_t N, typename T, typename... Args>
589  void parseConstructorArgs(BranchDefinition<T>&& def, Args&&... args)
590  {
591  if (def.nofBranches > 0) {
592  // number of branches set to 0 will skip the definition, this allows to
593  // dynamically disable branches, while all possible definitions can
594  // be specified at compile time
595  mInputs.insert(mInputs.end(), def.keys.begin(), def.keys.end());
596  mBranchNameOptions.emplace_back(def.optionKey, def.branchName);
597  mNofBranches += def.nofBranches;
598  } else {
599  // insert an empty placeholder
600  mBranchNameOptions.emplace_back("", "");
601  }
602  parseConstructorArgs<N + 1>(std::forward<Args>(args)...);
603  if constexpr (N == 0) {
604  mWriter = std::make_shared<WriterType>(nullptr, nullptr, mCustomClose, std::forward<BranchDefinition<T>>(def), std::forward<Args>(args)...);
605  }
606  }
