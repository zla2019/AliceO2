35 void EntropyEncoderSpec::run(ProcessingContext& pc)
36 {
37  auto cput = mTimer.CpuTime();
38  mTimer.Start(false);
39  auto compClusters = pc.inputs().get<gsl::span<o2::itsmft::CompClusterExt>>("compClusters");
40  auto pspan = pc.inputs().get<gsl::span<unsigned char>>("patterns");
41  auto rofs = pc.inputs().get<gsl::span<o2::itsmft::ROFRecord>>("ROframes");
42 
43  auto& buffer = pc.outputs().make<std::vector<o2::ctf::BufferType>>(Output{mOrigin, "CTFDATA", 0, Lifetime::Timeframe});
44  CTFCoder::encode(buffer, rofs, compClusters, pspan);
45  auto eeb = CTF::get(buffer.data()); // cast to container pointer
46  eeb->compactify(); // eliminate unnecessary padding
47  buffer.resize(eeb->size()); // shrink buffer to strictly necessary size
48  // eeb->print();
49  mTimer.Stop();
50  LOG(INFO) << "Created encoded data of size " << eeb->size() << " for " << mOrigin.as<std::string>() << " in " << mTimer.CpuTime() - cput << " s";
51 }

59 DataProcessorSpec getEntropyEncoderSpec(o2::header::DataOrigin orig)
60 {
61  std::vector<InputSpec> inputs;
62  inputs.emplace_back("compClusters", orig, "COMPCLUSTERS", 0, Lifetime::Timeframe);
63  inputs.emplace_back("patterns", orig, "PATTERNS", 0, Lifetime::Timeframe);
64  inputs.emplace_back("ROframes", orig, "CLUSTERSROF", 0, Lifetime::Timeframe);
65 
66  return DataProcessorSpec{
67  orig == o2::header::gDataOriginITS ? "its-entropy-encoder" : "mft-entropy-encoder",
68  inputs,
69  Outputs{{orig, "CTFDATA", 0, Lifetime::Timeframe}},
70  AlgorithmSpec{adaptFromTask<EntropyEncoderSpec>(orig)},
71  Options{}};
72 }
