21 BranchIterator::BranchIterator(TTree* tree, std::shared_ptr<arrow::ChunkedArray> col, std::shared_ptr<arrow::Field> field)
22 {
23  mField = field.get();
24  mBranchName = mField->name();
25 
26  mFieldType = mField->type()->id();
27  mChunks = col->chunks();
28  mNumberChuncs = mChunks.size();
29 
30  mLeaflistString = mBranchName;
31  mElementType = mFieldType;
32  mNumberElements = 1;
33  if (mFieldType == arrow::Type::type::FIXED_SIZE_LIST) {
34 
35  // element type
36  if (mField->type()->num_children() <= 0) {
37  LOGP(FATAL, "Field {} of type {} has no children!", mField->name(), mField->type()->ToString().c_str());
38  }
39  mElementType = mField->type()->child(0)->type()->id();
40 
41  // number of elements
42  mNumberElements = static_cast<const arrow::FixedSizeListType*>(mField->type().get())->list_size();
43  mLeaflistString += "[" + std::to_string(mNumberElements) + "]";
44  }
45 
46  // initialize the branch
47  mStatus = initBranch(tree);
48 
49  mCounterChunk = 0;
50  mStatus &= initDataBuffer(mCounterChunk);
51 }

74 bool BranchIterator::initBranch(TTree* tree)
75 {
76  // try to find branch in tree
77  mBranchPtr = tree->GetBranch(mBranchName.c_str());
78  if (mBranchPtr) {
79  return true;
80  }
81 
82  // create new branch of given data type
83  switch (mElementType) {
84  case arrow::Type::type::BOOL:
85  mLeaflistString += "/O";
86  break;
87  case arrow::Type::type::UINT8:
88  mLeaflistString += "/b";
89  break;
90  case arrow::Type::type::UINT16:
91  mLeaflistString += "/s";
92  break;
93  case arrow::Type::type::UINT32:
94  mLeaflistString += "/i";
95  break;
96  case arrow::Type::type::UINT64:
97  mLeaflistString += "/l";
98  break;
99  case arrow::Type::type::INT8:
100  mLeaflistString += "/B";
101  break;
102  case arrow::Type::type::INT16:
103  mLeaflistString += "/S";
104  break;
105  case arrow::Type::type::INT32:
106  mLeaflistString += "/I";
107  break;
108  case arrow::Type::type::INT64:
109  mLeaflistString += "/L";
110  break;
111  case arrow::Type::type::FLOAT:
112  mLeaflistString += "/F";
113  break;
114  case arrow::Type::type::DOUBLE:
115  mLeaflistString += "/D";
116  break;
117  default:
118  LOGP(FATAL, "Type {} not handled!", mElementType);
119  break;
120  }
121 
122  mBranchPtr = tree->Branch(mBranchName.c_str(), mBranchBuffer, mLeaflistString.c_str());
123  if (mBranchPtr) {
124  return true;
125  } else {
126  return false;
127  }
128 }

130 bool BranchIterator::initDataBuffer(Int_t ib)
131 {
132 
133  auto chunkToUse = mChunks.at(ib);
134  if (mFieldType == arrow::Type::type::FIXED_SIZE_LIST) {
135  chunkToUse = std::dynamic_pointer_cast<arrow::FixedSizeListArray>(chunkToUse)->values();
136  }
137 
138  // reset actual row number
139  mCounterRow = 0;
140 
141  // get next chunk of given data type mElementType
142  switch (mElementType) {
143  case arrow::Type::type::BOOL:
144  if (!mVariable_o) {
145  mVariable_o = new bool(mNumberElements);
146  }
147  mArray_o = std::dynamic_pointer_cast<arrow::BooleanArray>(chunkToUse);
148  for (int ii = 0; ii < mNumberElements; ii++) {
149  mVariable_o[ii] = (bool)mArray_o->Value(ii);
150  }
151  mValueBuffer = (void*)mVariable_o;
152  break;
153  case arrow::Type::type::UINT8:
154  mVariable_ub = (uint8_t*)std::dynamic_pointer_cast<arrow::UInt8Array>(chunkToUse)->raw_values();
155  mValueBuffer = (void*)mVariable_ub;
156  break;
157  case arrow::Type::type::UINT16:
158  mVariable_us = (uint16_t*)std::dynamic_pointer_cast<arrow::UInt16Array>(chunkToUse)->raw_values();
159  mValueBuffer = (void*)mVariable_us;
160  break;
161  case arrow::Type::type::UINT32:
162  mVariable_ui = (uint32_t*)std::dynamic_pointer_cast<arrow::UInt32Array>(chunkToUse)->raw_values();
163  mValueBuffer = (void*)mVariable_ui;
164  break;
165  case arrow::Type::type::UINT64:
166  mVariable_ul = (uint64_t*)std::dynamic_pointer_cast<arrow::UInt64Array>(chunkToUse)->raw_values();
167  mValueBuffer = (void*)mVariable_ul;
168  break;
169  case arrow::Type::type::INT8:
170  mVariable_b = (int8_t*)std::dynamic_pointer_cast<arrow::Int8Array>(chunkToUse)->raw_values();
171  mValueBuffer = (void*)mVariable_b;
172  break;
173  case arrow::Type::type::INT16:
174  mVariable_s = (int16_t*)std::dynamic_pointer_cast<arrow::Int16Array>(chunkToUse)->raw_values();
175  mValueBuffer = (void*)mVariable_s;
176  break;
177  case arrow::Type::type::INT32:
178  mVariable_i = (int32_t*)std::dynamic_pointer_cast<arrow::Int32Array>(chunkToUse)->raw_values();
179  mValueBuffer = (void*)mVariable_i;
180  break;
181  case arrow::Type::type::INT64:
182  mVariable_l = (int64_t*)std::dynamic_pointer_cast<arrow::Int64Array>(chunkToUse)->raw_values();
183  mValueBuffer = (void*)mVariable_l;
184  break;
185  case arrow::Type::type::FLOAT:
186  mVariable_f = (Float_t*)std::dynamic_pointer_cast<arrow::FloatArray>(chunkToUse)->raw_values();
187  mValueBuffer = (void*)mVariable_f;
188  break;
189  case arrow::Type::type::DOUBLE:
190  mVariable_d = (double*)std::dynamic_pointer_cast<arrow::DoubleArray>(chunkToUse)->raw_values();
191  mValueBuffer = (void*)mVariable_d;
192  default:
193  break;
194  LOGP(FATAL, "Type {} not handled!", mElementType);
195  break;
196  }
197  mBranchPtr->SetAddress(mValueBuffer);
198 
199  // reset number of rows mNumberRows and row counter mCounterRow
200  mNumberRows = mChunks.at(ib)->length();
201 
202  return true;
203 }

205 bool BranchIterator::push()
206 {
207  // increment row counter
208  mCounterRow++;
209 
210  // mCounterChunk and mCounterRow contain the current chunk and row
211  // return the next element if available
212  if (mCounterRow >= mNumberRows) {
213  mCounterChunk++;
214  if (mCounterChunk < mNumberChuncs) {
215  initDataBuffer(mCounterChunk);
216  } else {
217  // end of data buffer reached
218  return false;
219  }
220  } else {
221  switch (mElementType) {
222  case arrow::Type::type::BOOL:
223  for (int ii = 0; ii < mNumberElements; ii++) {
224  mVariable_o[ii] = (bool)mArray_o->Value(mCounterRow * mNumberElements + ii);
225  }
226  mValueBuffer = (void*)mVariable_o;
227  break;
228  case arrow::Type::type::UINT8:
229  mVariable_ub += mNumberElements;
230  mValueBuffer = (void*)mVariable_ub;
231  break;
232  case arrow::Type::type::UINT16:
233  mVariable_us += mNumberElements;
234  mValueBuffer = (void*)mVariable_us;
235  break;
236  case arrow::Type::type::UINT32:
237  mVariable_ui += mNumberElements;
238  mValueBuffer = (void*)mVariable_ui;
239  break;
240  case arrow::Type::type::UINT64:
241  mVariable_ul += mNumberElements;
242  mValueBuffer = (void*)mVariable_ul;
243  break;
244  case arrow::Type::type::INT8:
245  mVariable_b += mNumberElements;
246  mValueBuffer = (void*)mVariable_b;
247  break;
248  case arrow::Type::type::INT16:
249  mVariable_s += mNumberElements;
250  mValueBuffer = (void*)mVariable_s;
251  break;
252  case arrow::Type::type::INT32:
253  mVariable_i += mNumberElements;
254  mValueBuffer = (void*)mVariable_i;
255  break;
256  case arrow::Type::type::INT64:
257  mVariable_l += mNumberElements;
258  mValueBuffer = (void*)mVariable_l;
259  break;
260  case arrow::Type::type::FLOAT:
261  mVariable_f += mNumberElements;
262  mValueBuffer = (void*)mVariable_f;
263  break;
264  case arrow::Type::type::DOUBLE:
265  mVariable_d += mNumberElements;
266  mValueBuffer = (void*)mVariable_d;
267  break;
268  default:
269  LOGP(FATAL, "Type {} not handled!", mElementType);
270  break;
271  }
272  }
273  mBranchPtr->SetAddress(mValueBuffer);
274 
275  return true;
276 }

326 TTree* TableToTree::process()
327 {
328 
329  bool togo = true;
330  while (togo) {
331  // fill the tree
332  mTreePtr->Fill();
333 
334  // update the branches
335  for (auto brit : mBranchIterators) {
336  togo &= brit->push();
337  }
338  }
339  mTreePtr->Write();
340 
341  return mTreePtr;
342 }

373 // is used in TreeToTable
374 ColumnIterator::ColumnIterator(TTreeReader* reader, const char* colname)
375 {
376 
377  // find branch
378  auto tree = reader->GetTree();
379  if (!tree) {
380  LOGP(FATAL, "Can not locate tree!");
381  return;
382  }
383  //tree->Print();
384  auto br = tree->GetBranch(colname);
385  if (!br) {
386  LOGP(FATAL, "Can not locate branch {}", colname);
387  return;
388  }
389  mColumnName = colname;
390 
391  // type of the branch elements
392  TClass* cl;
393  br->GetExpectedType(cl, mElementType);
394 
395  // currently only single-value or single-array branches are accepted
396  // thus of the form e.g. alpha/D or alpha[5]/D
397  // check if this is a single-value or single-array branch
398  mNumberElements = 1;
399  std::string branchTitle = br->GetTitle();
400  Int_t pos0 = branchTitle.find("[");
401  Int_t pos1 = branchTitle.find("]");
402  if (pos0 > 0 && pos1 > 0) {
403  mNumberElements = atoi(branchTitle.substr(pos0 + 1, pos1 - pos0 - 1).c_str());
404  }
405 
406  // initialize the TTreeReaderValue<T> / TTreeReaderArray<T>
407  // the corresponding arrow::TBuilder
408  // the column field
409  // the TTreeReaderValue is incremented by reader->Next()
410  // switch according to mElementType
411  mStatus = true;
412 
413  if (mNumberElements == 1) {
414  switch (mElementType) {
415  case EDataType::kBool_t:
416  mReaderValue_o = new TTreeReaderValue<bool>(*reader, mColumnName);
417  MAKE_FIELD_AND_BUILDER(bool, 1, mTableBuilder_o);
418  break;
419  case EDataType::kUChar_t:
420  mReaderValue_ub = new TTreeReaderValue<uint8_t>(*reader, mColumnName);
421  MAKE_FIELD_AND_BUILDER(uint8_t, 1, mTableBuilder_ub);
422  break;
423  case EDataType::kUShort_t:
424  mReaderValue_us = new TTreeReaderValue<uint16_t>(*reader, mColumnName);
425  MAKE_FIELD_AND_BUILDER(uint16_t, 1, mTableBuilder_us);
426  break;
427  case EDataType::kUInt_t:
428  mReaderValue_ui = new TTreeReaderValue<uint32_t>(*reader, mColumnName);
429  MAKE_FIELD_AND_BUILDER(uint32_t, 1, mTableBuilder_ui);
430  break;
431  case EDataType::kULong64_t:
432  mReaderValue_ul = new TTreeReaderValue<uint64_t>(*reader, mColumnName);
433  MAKE_FIELD_AND_BUILDER(uint64_t, 1, mTableBuilder_ul);
434  break;
435  case EDataType::kChar_t:
436  mReaderValue_b = new TTreeReaderValue<int8_t>(*reader, mColumnName);
437  MAKE_FIELD_AND_BUILDER(int8_t, 1, mTableBuilder_b);
438  break;
439  case EDataType::kShort_t:
440  mReaderValue_s = new TTreeReaderValue<int16_t>(*reader, mColumnName);
441  MAKE_FIELD_AND_BUILDER(int16_t, 1, mTableBuilder_s);
442  break;
443  case EDataType::kInt_t:
444  mReaderValue_i = new TTreeReaderValue<int32_t>(*reader, mColumnName);

549 void ColumnIterator::push()
550 {
551  arrow::Status stat;
552 
553  // switch according to mElementType
554  if (mNumberElements == 1) {
555  switch (mElementType) {
556  case EDataType::kBool_t:
557  stat = mTableBuilder_o->Append((bool)**mReaderValue_o);
558  break;
559  case EDataType::kUChar_t:
560  stat = mTableBuilder_ub->Append(**mReaderValue_ub);
561  break;
562  case EDataType::kUShort_t:
563  stat = mTableBuilder_us->Append(**mReaderValue_us);
564  break;
565  case EDataType::kUInt_t:
566  stat = mTableBuilder_ui->Append(**mReaderValue_ui);
567  break;
568  case EDataType::kULong64_t:
569  stat = mTableBuilder_ul->Append(**mReaderValue_ul);
570  break;
571  case EDataType::kChar_t:
572  stat = mTableBuilder_b->Append(**mReaderValue_b);
573  break;
574  case EDataType::kShort_t:
575  stat = mTableBuilder_s->Append(**mReaderValue_s);
576  break;
577  case EDataType::kInt_t:
578  stat = mTableBuilder_i->Append(**mReaderValue_i);
579  break;
580  case EDataType::kLong64_t:
581  stat = mTableBuilder_l->Append(**mReaderValue_l);
582  break;
583  case EDataType::kFloat_t:
584  stat = mTableBuilder_f->Append(**mReaderValue_f);
585  break;
586  case EDataType::kDouble_t:
587  stat = mTableBuilder_d->Append(**mReaderValue_d);
588  break;
589  default:
590  LOGP(FATAL, "Type {} not handled!", mElementType);
591  break;
592  }
593  } else {
594  stat = mTableBuilder_list->AppendValues(1);
595  switch (mElementType) {
596  case EDataType::kBool_t:
597  stat &= mTableBuilder_o->AppendValues((uint8_t*)&((*mReaderArray_o)[0]), mNumberElements);
598  break;
599  case EDataType::kUChar_t:
600  stat &= mTableBuilder_ub->AppendValues(&((*mReaderArray_ub)[0]), mNumberElements);
601  break;
602  case EDataType::kUShort_t:
603  stat &= mTableBuilder_us->AppendValues(&((*mReaderArray_us)[0]), mNumberElements);
604  break;
605  case EDataType::kUInt_t:
606  stat &= mTableBuilder_ui->AppendValues(&((*mReaderArray_ui)[0]), mNumberElements);
607  break;
608  case EDataType::kULong64_t:
609  stat &= mTableBuilder_ul->AppendValues(&((*mReaderArray_ul)[0]), mNumberElements);
610  break;
611  case EDataType::kChar_t:
612  stat &= mTableBuilder_b->AppendValues(&((*mReaderArray_b)[0]), mNumberElements);
613  break;
614  case EDataType::kShort_t:
615  stat &= mTableBuilder_s->AppendValues(&((*mReaderArray_s)[0]), mNumberElements);
616  break;
617  case EDataType::kInt_t:
618  stat &= mTableBuilder_i->AppendValues(&((*mReaderArray_i)[0]), mNumberElements);
619  break;
620  case EDataType::kLong64_t:
621  stat &= mTableBuilder_l->AppendValues(&((*mReaderArray_l)[0]), mNumberElements);
622  break;
623  case EDataType::kFloat_t:
624  stat &= mTableBuilder_f->AppendValues(&((*mReaderArray_f)[0]), mNumberElements);
