45  TOFChannelData(int nb, float r, CalibTOFapi* cta) : mNBins(nb), mRange(r), mCalibTOFapi(cta)
46  {
47  if (r <= 0. || nb < 1) {
48  throw std::runtime_error("Wrong initialization of the histogram");
49  }
50  mV2Bin = mNBins / (2 * mRange);
51  for (int isect = 0; isect < 18; isect++) {
52  mHisto[isect] = boost::histogram::make_histogram(boost::histogram::axis::regular<>(mNBins, -mRange, mRange, "t-texp"),
53  boost::histogram::axis::integer<>(0, o2::tof::Geo::NPADSXSECTOR, "channel index in sector" + std::to_string(isect))); // bin is defined as [low, high[
54  }
55  mEntries.resize(o2::tof::Geo::NCHANNELS, 0);
56  }
