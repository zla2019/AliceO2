34 {
35  auto cput = mTimer.CpuTime();
36  mTimer.Start(false);
37 
38  auto buff = pc.inputs().get<gsl::span<o2::ctf::BufferType>>("ctf");
39 
40  auto& compclusters = pc.outputs().make<std::vector<char>>(OutputRef{"output"});
41  const auto ctfImage = o2::tpc::CTF::getImage(buff.data());
42  CTFCoder::decode(ctfImage, compclusters);
43 
44  mTimer.Stop();
45  LOG(INFO) << "Decoded " << buff.size() * sizeof(o2::ctf::BufferType) << " encoded bytes to "
46  << compclusters.size() << " bytes in" << mTimer.CpuTime() - cput << "\n";
47 }
