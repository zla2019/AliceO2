30 void EntropyEncoderSpec::run(ProcessingContext& pc)
31 {
32  CompressedClusters clusters;
33 
34  if (mFromFile) {
35  auto tmp = pc.inputs().get<CompressedClustersROOT*>("input");
36  if (tmp == nullptr) {
37  LOG(ERROR) << "invalid input";
38  return;
39  }
40  clusters = *tmp;
41  } else {
42  auto tmp = pc.inputs().get<CompressedClustersFlat*>("input");
43  if (tmp == nullptr) {
44  LOG(ERROR) << "invalid input";
45  return;
46  }
47  clusters = *tmp;
48  }
49  auto cput = mTimer.CpuTime();
50  mTimer.Start(false);
51 
52  auto& buffer = pc.outputs().make<std::vector<o2::ctf::BufferType>>(Output{"TPC", "CTFDATA", 0, Lifetime::Timeframe});
53  CTFCoder::encode(buffer, clusters);
54  auto encodedBlocks = CTF::get(buffer.data()); // cast to container pointer
55  encodedBlocks->compactify(); // eliminate unnecessary padding
56  buffer.resize(encodedBlocks->size()); // shrink buffer to strictly necessary size
57  // encodedBlocks->print();
58  mTimer.Stop();
59  LOG(INFO) << "Created encoded data of size " << encodedBlocks->size() << " for TPC in " << mTimer.CpuTime() - cput << " s";
60 }
