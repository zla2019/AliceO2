40 void CTFCoder::compress(CompressedClusters& cc,
41  const gsl::span<const ROFRecord>& rofRecVec,
42  const gsl::span<const CompClusterExt>& cclusVec,
43  const gsl::span<const unsigned char>& pattVec)
44 {
45  // store in the header the orbit of 1st ROF
46  cc.clear();
47  if (!rofRecVec.size()) {
48  return;
49  }
50  const auto& rofRec0 = rofRecVec[0];
51  cc.header.nROFs = rofRecVec.size();
52  cc.header.firstOrbit = rofRec0.getBCData().orbit;
53  cc.header.firstBC = rofRec0.getBCData().bc;
54  cc.header.nPatternBytes = pattVec.size();
55  cc.header.nClusters = cclusVec.size();
56  cc.header.nChips = 0; // this is the version with chipInc stored once per new chip
57 
58  cc.firstChipROF.resize(cc.header.nROFs);
59  cc.bcIncROF.resize(cc.header.nROFs);
60  cc.orbitIncROF.resize(cc.header.nROFs);
61  cc.nclusROF.resize(cc.header.nROFs);
62  //
63  cc.row.resize(cc.header.nClusters);
64  cc.colInc.resize(cc.header.nClusters);
65  // cc.chipInc.resize(cc.header.nClusters); // this is the version with chipInc stored for every pixel
66  cc.chipInc.reserve(1000); // this is the version with chipInc stored once per new chip
67  cc.chipMul.reserve(1000); // this is the version with chipInc stored once per new chip
68  cc.pattID.resize(cc.header.nClusters);
69  cc.pattMap.resize(cc.header.nPatternBytes);
70 
71  uint16_t prevBC = cc.header.firstBC;
72  uint32_t prevOrbit = cc.header.firstOrbit;
73 
74  for (uint32_t irof = 0; irof < cc.header.nROFs; irof++) {
75  const auto& rofRec = rofRecVec[irof];
76  const auto& intRec = rofRec.getBCData();
77  if (intRec.orbit == prevOrbit) {
78  cc.orbitIncROF[irof] = 0;
79  cc.bcIncROF[irof] = intRec.bc - prevBC; // store increment of BC if in the same orbit
80  } else {
81  cc.orbitIncROF[irof] = intRec.orbit - prevOrbit;
82  cc.bcIncROF[irof] = intRec.bc; // otherwise, store absolute bc
83  prevOrbit = intRec.orbit;
84  }
85  prevBC = intRec.bc;
86  auto ncl = rofRec.getNEntries();
87  cc.nclusROF[irof] = ncl;
88  if (!ncl) { // no hits data for this ROF
89  cc.firstChipROF[irof] = 0;
90  continue;
91  }
92  cc.firstChipROF[irof] = cclusVec[rofRec.getFirstEntry()].getChipID();
93  int icl = rofRec.getFirstEntry(), iclMax = icl + ncl;
94 
95  uint16_t prevChip = cc.firstChipROF[irof], prevCol = 0;
96  if (icl != iclMax) { // there still clusters to store
97  cc.chipMul.push_back(0);
98  cc.chipInc.push_back(0);
99  }
100  for (; icl < iclMax; icl++) { // clusters within a chip are stored in increasing column number
101  const auto& cl = cclusVec[icl];
102  cc.row[icl] = cl.getRow(); // row is practically random, store it as it is
103  cc.pattID[icl] = cl.getPatternID();
104  if (cl.getChipID() == prevChip) { // for the same chip store column increment
105  // cc.chipInc[icl] = 0; // this is the version with chipInc stored for every pixel
106  cc.chipMul.back()++; // this is the version with chipInc stored once per new chip
107  cc.colInc[icl] = cl.getCol() - prevCol;
108  prevCol = cl.getCol();
109  } else { // for new chips store chipID increment and abs. column
110  // cc.chipInc[icl] = cl.getChipID() - prevChip; // this is the version with chipInc stored for every pixel
111  cc.chipInc.push_back(cl.getChipID() - prevChip); // this is the version with chipInc stored once per new chip
112  cc.chipMul.push_back(1); // this is the version with chipInc stored once per new chip
113  prevCol = cc.colInc[icl] = cl.getCol();
114  prevChip = cl.getChipID();
115  }
116  }
117  }
118  cc.header.nChips = cc.chipMul.size();
119  // store explicit patters as they are
120  memcpy(cc.pattMap.data(), pattVec.data(), cc.header.nPatternBytes); // RSTODO: do we need this?
121 }
