102 template <typename IT>
103 SymbolStatistics::SymbolStatistics(const IT begin, const IT end, size_t range) : mMin(0), mMax(0), mNUsedAlphabetSymbols(0), mMessageLength(0), mFrequencyTable(), mCumulativeFrequencyTable()
104 {
105  LOG(trace) << "start building symbol statistics";
106  RANSTimer t;
107  t.start();
108  if (begin == end) {
109  LOG(warning) << "Passed empty message to " << __func__;
110  return;
111  }
112 
113  buildFrequencyTable(begin, end, range);
114 
115  for (auto i : mFrequencyTable) {
116  if (i > 0) {
117  mNUsedAlphabetSymbols++;
118  }
119  }
120 
121  buildCumulativeFrequencyTable();
122 
123  mMessageLength = mCumulativeFrequencyTable.back();
124 
125  assert(mFrequencyTable.size() > 0);
126  assert(mCumulativeFrequencyTable.size() > 1);
127  assert(mCumulativeFrequencyTable.size() - mFrequencyTable.size() == 1);
128 
129  t.stop();
130  LOG(debug1) << __func__ << " inclusive time (ms): " << t.getDurationMS();
131 
132 // advanced diagnostics in debug builds
133 #if !defined(NDEBUG)
134  [&]() {
135  const uint messageRange = [&]() -> uint {
136  if (range > 0) {
137  return range;
138  } else if (mMax - mMin == 0) {
139  return 1;
140  } else {
141  return std::ceil(std::log2(mMax - mMin));
142  } }();
143 
144  double entropy = 0;
145  for (auto frequency : mFrequencyTable) {
146  if (frequency > 0) {
147  const double p = (frequency * 1.0) / mMessageLength;
148  entropy -= p * std::log2(p);
149  }
150  }
151 
152  LOG(debug2) << "messageProperties: {"
153  << "numSymbols: " << mMessageLength << ", "
154  << "alphabetRange: " << messageRange << ", "
155  << "alphabetSize: " << mNUsedAlphabetSymbols << ", "
156  << "minSymbol: " << mMin << ", "
157  << "maxSymbol: " << mMax << ", "
158  << "entropy: " << entropy << ", "
159  << "bufferSizeB: " << mMessageLength * sizeof(typename std::iterator_traits<IT>::value_type) << ", "
160  << "actualSizeB: " << static_cast<int>(mMessageLength * messageRange / 8) << ", "
161  << "entropyMessageB: " << static_cast<int>(std::ceil(entropy * mMessageLength / 8)) << "}";
162 
163  LOG(debug2) << "SymbolStatistics: {"
164  << "entries: " << mFrequencyTable.size() << ", "
165  << "frequencyTableSizeB: " << mFrequencyTable.size() * sizeof(std::decay_t<decltype(mFrequencyTable)>::value_type) << ", "
166  << "CumulativeFrequencyTableSizeB: " << mCumulativeFrequencyTable.size() * sizeof(std::decay_t<decltype(mCumulativeFrequencyTable)>::value_type) << "}";
167  }();
168 #endif
169 
170  LOG(trace) << "done building symbol statistics";
171 }

188 template <typename IT>
189 void SymbolStatistics::buildFrequencyTable(const IT begin, const IT end,
190  size_t range)
191 {
192  LOG(trace) << "start building frequency table";
193  // find min_ and max_
194  const auto minmax = std::minmax_element(begin, end);
195 
196  if (range > 0) {
197  mMin = 0;
198  mMax = (1 << range) - 1;
199 
200  // do checks
201  if (static_cast<unsigned int>(mMin) > *minmax.first) {
202  throw std::runtime_error("min of data too small for specified range");
203  }
204 
205  if (static_cast<unsigned int>(mMax) < *minmax.second) {
206  throw std::runtime_error("max of data too big for specified range");
207  }
208  } else {
209  mMin = *minmax.first;
210  mMax = *minmax.second;
211  }
212  assert(mMax >= mMin);
213 
214  mFrequencyTable.resize(std::abs(mMax - mMin) + 1, 0);
215 
216  for (IT it = begin; it != end; it++) {
217  mFrequencyTable[*it - mMin]++;
218  }
219  LOG(trace) << "done building frequency table";
220 }
