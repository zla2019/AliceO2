56 template <typename T>
57 constexpr auto selectArrowType()
58 {
59  if constexpr (std::is_same_v<T, int>) {
60  return atype::INT32;
61  } else if constexpr (std::is_same_v<T, bool>) {
62  return atype::BOOL;
63  } else if constexpr (std::is_same_v<T, float>) {
64  return atype::FLOAT;
65  } else if constexpr (std::is_same_v<T, double>) {
66  return atype::DOUBLE;
67  } else if constexpr (std::is_same_v<T, int8_t>) {
68  return atype::INT8;
69  } else if constexpr (std::is_same_v<T, uint16_t>) {
70  return atype::INT16;
71  } else if constexpr (std::is_same_v<T, uint8_t>) {
72  return atype::UINT8;
73  } else {
74  return atype::NA;
75  }
76  O2_BUILTIN_UNREACHABLE();
77 }
