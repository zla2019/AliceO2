126 BOOST_AUTO_TEST_CASE(TestGandivaTreeCreation)
127 {
128  Projector pze = o2::aod::track::Pze::Projector();
129  auto pzspecs = createOperations(std::move(pze));
130  BOOST_REQUIRE_EQUAL(pzspecs[0].left, (DatumSpec{std::string{"fTgl"}, atype::FLOAT}));
131  BOOST_REQUIRE_EQUAL(pzspecs[0].right, (DatumSpec{1u, atype::FLOAT}));
132  BOOST_REQUIRE_EQUAL(pzspecs[0].result, (DatumSpec{0u, atype::FLOAT}));
133 
134  BOOST_REQUIRE_EQUAL(pzspecs[1].left, (DatumSpec{LiteralNode::var_t{1.f}, atype::FLOAT}));
135  BOOST_REQUIRE_EQUAL(pzspecs[1].right, (DatumSpec{std::string{"fSigned1Pt"}, atype::FLOAT}));
136  BOOST_REQUIRE_EQUAL(pzspecs[1].result, (DatumSpec{1u, atype::FLOAT}));
137  auto infield1 = o2::aod::track::Signed1Pt::asArrowField();
138  auto infield2 = o2::aod::track::Tgl::asArrowField();
139  auto resfield = o2::aod::track::Pze::asArrowField();
140  auto schema = std::make_shared<arrow::Schema>(std::vector{infield1, infield2, resfield});
141  auto gandiva_tree = createExpressionTree(pzspecs, schema);
142 
143  auto gandiva_expression = makeExpression(gandiva_tree, resfield);
144  BOOST_CHECK_EQUAL(gandiva_expression->ToString(), "float multiply((float) fTgl, float divide((const float) 1 raw(3f800000), (float) fSigned1Pt))");
145  auto projector = createProjector(schema, pzspecs, resfield);
