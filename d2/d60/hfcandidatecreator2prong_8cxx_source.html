30 struct HFCandidateCreator2Prong {
31  Produces<aod::HfCandProng2> hfcandprong2;
32  Configurable<bool> b_dovalplots{"b_dovalplots", true, "do validation plots"};
33  OutputObj<TH1F> hvtx_x_out{TH1F("hvtx_x", "2-track vtx", 100, -0.1, 0.1)};
34  OutputObj<TH1F> hvtx_y_out{TH1F("hvtx_y", "2-track vtx", 100, -0.1, 0.1)};
35  OutputObj<TH1F> hvtx_z_out{TH1F("hvtx_z", "2-track vtx", 100, -0.1, 0.1)};
36  OutputObj<TH1F> hmass2{TH1F("hmass2", "2-track inv mass", 500, 0, 5.0)};
37  Configurable<double> d_bz{"d_bz", 5.0, "bz field"};
38  Configurable<bool> b_propdca{"b_propdca", true,
39  "create tracks version propagated to PCA"};
40  Configurable<double> d_maxr{"d_maxr", 200, "reject PCA's above this radius"};
41  Configurable<double> d_maxdzini{"d_maxdzini", 4,
42  "reject (if>0) PCA candidate if tracks DZ exceeds threshold"};
43  Configurable<double> d_minparamchange{"d_minparamchange", 1e-3,
44  "stop iterations if largest change of any X is smaller than this"};
45  Configurable<double> d_minrelchi2change{"d_minrelchi2change", 0.9,
46  "stop iterations is chi2/chi2old > this"};
47  void process(aod::Collision const& collision,
48  aod::HfTrackIndexProng2 const& hftrackindexprong2s,
49  soa::Join<aod::Tracks, aod::TracksCov, aod::TracksExtra> const& tracks)
50  {
51  o2::vertexing::DCAFitterN<2> df;
52  df.setBz(d_bz);
53  df.setPropagateToPCA(b_propdca);
54  df.setMaxR(d_maxr);
55  df.setMaxDZIni(d_maxdzini);
56  df.setMinParamChange(d_minparamchange);
57  df.setMinRelChi2Change(d_minrelchi2change);
58 
59  for (auto& hfpr2 : hftrackindexprong2s) {
60  float x_p1 = hfpr2.index0().x();
61  float alpha_p1 = hfpr2.index0().alpha();
62  std::array<float, 5> arraypar_p1 = {hfpr2.index0().y(), hfpr2.index0().z(), hfpr2.index0().snp(),
63  hfpr2.index0().tgl(), hfpr2.index0().signed1Pt()};
64  std::array<float, 15> covpar_p1 = {hfpr2.index0().cYY(), hfpr2.index0().cZY(), hfpr2.index0().cZZ(),
65  hfpr2.index0().cSnpY(), hfpr2.index0().cSnpZ(),
66  hfpr2.index0().cSnpSnp(), hfpr2.index0().cTglY(), hfpr2.index0().cTglZ(),
67  hfpr2.index0().cTglSnp(), hfpr2.index0().cTglTgl(),
68  hfpr2.index0().c1PtY(), hfpr2.index0().c1PtZ(), hfpr2.index0().c1PtSnp(),
69  hfpr2.index0().c1PtTgl(), hfpr2.index0().c1Pt21Pt2()};
70  o2::track::TrackParCov trackparvar_p1(x_p1, alpha_p1, arraypar_p1, covpar_p1);
71 
72  float x_n1 = hfpr2.index1().x();
73  float alpha_n1 = hfpr2.index1().alpha();
74  std::array<float, 5> arraypar_n1 = {hfpr2.index1().y(), hfpr2.index1().z(), hfpr2.index1().snp(),
75  hfpr2.index1().tgl(), hfpr2.index1().signed1Pt()};
76  std::array<float, 15> covpar_n1 = {hfpr2.index1().cYY(), hfpr2.index1().cZY(), hfpr2.index1().cZZ(),
77  hfpr2.index1().cSnpY(), hfpr2.index1().cSnpZ(),
78  hfpr2.index1().cSnpSnp(), hfpr2.index1().cTglY(), hfpr2.index1().cTglZ(),
79  hfpr2.index1().cTglSnp(), hfpr2.index1().cTglTgl(),
80  hfpr2.index1().c1PtY(), hfpr2.index1().c1PtZ(), hfpr2.index1().c1PtSnp(),
81  hfpr2.index1().c1PtTgl(), hfpr2.index1().c1Pt21Pt2()};
82  o2::track::TrackParCov trackparvar_n1(x_n1, alpha_n1, arraypar_n1, covpar_n1);
83  df.setUseAbsDCA(true);
84  int nCand = df.process(trackparvar_p1, trackparvar_n1);
85  if (nCand == 0)
86  continue;
87  const auto& vtx = df.getPCACandidate();
88  std::array<float, 3> pvec0;
89  std::array<float, 3> pvec1;
90  df.getTrack(0).getPxPyPzGlo(pvec0);
91  df.getTrack(1).getPxPyPzGlo(pvec1);
92  float masspion = 0.140;
93  float masskaon = 0.494;
94  float mass_ = sqrt(invmass2prongs2(pvec0[0], pvec0[1],
95  pvec0[2], masspion,
96  pvec1[0], pvec1[1],
97  pvec1[2], masskaon));
98  float masssw_ = sqrt(invmass2prongs2(pvec0[0], pvec0[1],
99  pvec0[2], masskaon,
100  pvec1[0], pvec1[1],
101  pvec1[2], masspion));
102  hfcandprong2(collision.posX(), collision.posY(), collision.posZ(),
103  pvec0[0], pvec0[1], pvec0[2], pvec1[0], pvec1[1], pvec1[2],
104  vtx[0], vtx[1], vtx[2], mass_, masssw_);
105  if (b_dovalplots == true) {
106  hvtx_x_out->Fill(vtx[0]);
107  hvtx_y_out->Fill(vtx[1]);
108  hvtx_z_out->Fill(vtx[2]);
109  hmass2->Fill(mass_);
110  hmass2->Fill(masssw_);
111  }
112  }
113  }
114 };
