59 unsigned int Tracklet::getTrackletWord() const
60 {
61 
62  int rndAdd = 0;
63  int decPlaces = 5; // must be larger than 1 or change the following code
64  // if (decPlaces > 1)
65  rndAdd = (1 << (decPlaces - 1)) + 1;
66  uint32_t trackletWord;
67  unsigned int pid, padrow, slope, offset;
68 
69  pid = ((int)mPID) & 0xff; //8bits
70  padrow = ((int)getZ()) & 0xf; //4bits
71  slope = ((int)mdY) & 0x7ff; //7bits
72  offset = ((int)mY) & 0x1fff; //13bits
73  //mMCMT[cpu] = (pid << 24) | (padrow << 20) | (slope << 13) | offset;
74  trackletWord = (pid << 24) | (padrow << 20) | (slope << 13) | offset;
75 
76  return trackletWord;
77 }
