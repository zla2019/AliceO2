109 inline auto findPad(std::vector<PadOriginal>& pads, double x, double y, double minCharge)
110 {
113 
114  auto match = [x, y, minCharge](const PadOriginal& pad) {
115  return pad.charge() >= minCharge && TMath::Abs(pad.x() - x) < 1.e-3 && TMath::Abs(pad.y() - y) < 1.e-3;
116  };
117 
118  auto itPad = std::find_if(pads.begin(), pads.end(), match);
119 
120  if (itPad == pads.end()) {
121  throw std::runtime_error("Pad not found");
122  }
123 
124  return itPad;
125 }

128 inline bool areOverlapping(const PadOriginal& pad1, const PadOriginal& pad2, double precision)
129 {
133 
134  if (pad1.x() - pad1.dx() > pad2.x() + pad2.dx() - precision ||
135  pad1.x() + pad1.dx() < pad2.x() - pad2.dx() + precision ||
136  pad1.y() - pad1.dy() > pad2.y() + pad2.dy() - precision ||
137  pad1.y() + pad1.dy() < pad2.y() - pad2.dy() + precision) {
138  return false;
139  }
140 
141  return true;
142 }
