64  for (auto& track : tracksBarrel) {

72  track.tpcNClsShared(), track.tpcChi2NCl(),
73  track.tpcSignal(), track.trdSignal(), track.tofSignal(),
74  track.trdChi2(), track.tofChi2(),

77  }
