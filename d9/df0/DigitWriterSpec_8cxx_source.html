39 DataProcessorSpec getDigitWriterSpec(bool mctruth, o2::header::DataOrigin detOrig, o2::detectors::DetID detId)
40 {
41  std::string detStr = o2::detectors::DetID::getName(detId);
42  std::string detStrL = detStr;
43  std::transform(detStrL.begin(), detStrL.end(), detStrL.begin(), ::tolower);
44  auto logger = [](std::vector<o2::itsmft::Digit> const& inDigits) {
45  LOG(INFO) << "RECEIVED DIGITS SIZE " << inDigits.size();
46  };
47  return MakeRootTreeWriterSpec((detStr + "DigitWriter").c_str(),
48  (detStrL + "digits.root").c_str(),
49  MakeRootTreeWriterSpec::TreeAttributes{"o2sim", "Digits tree"},
50  BranchDefinition<MCCont>{InputSpec{"digitsMCTR", detOrig, "DIGITSMCTR", 0},
51  (detStr + "DigitMCTruth").c_str(),
52  (mctruth ? 1 : 0)},
53  BranchDefinition<std::vector<itsmft::MC2ROFRecord>>{InputSpec{"digitsMC2ROF", detOrig, "DIGITSMC2ROF", 0},
54  (detStr + "DigitMC2ROF").c_str(),
55  (mctruth ? 1 : 0)},
56  BranchDefinition<std::vector<itsmft::Digit>>{InputSpec{"digits", detOrig, "DIGITS", 0},
57  (detStr + "Digit").c_str(),
58  logger},
59  BranchDefinition<std::vector<itsmft::ROFRecord>>{InputSpec{"digitsROF", detOrig, "DIGITSROF", 0},
60  (detStr + "DigitROF").c_str()})();
61 }
