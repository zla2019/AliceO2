25 struct EvSelParameters {
26  // time-of-flight offset
27  float fV0ADist = 329.00 / TMath::Ccgs() * 1e9; // ns
28  float fV0CDist = 87.15 / TMath::Ccgs() * 1e9; // ns
29 
30  float fFDADist = (1695.30 + 1698.04) / TMath::Ccgs() * 1e9; // ns
31  float fFDCDist = (1952.90 + 1955.90) / TMath::Ccgs() * 1e9; // ns
32 
33  // beam-beam and beam-gas windows
34  float fV0ABBlower = +fV0ADist - 9.5; // ns
35  float fV0ABBupper = +fV0ADist + 22.5; // ns
36  float fV0ABGlower = -fV0ADist - 2.5; // ns
37  float fV0ABGupper = -fV0ADist + 5.0; // ns
38  float fV0CBBlower = +fV0CDist - 2.5; // ns
39  float fV0CBBupper = +fV0CDist + 22.5; // ns
40  float fV0CBGlower = -fV0CDist - 2.5; // ns
41  float fV0CBGupper = -fV0CDist + 2.5; // ns
42 
43  float fFDABBlower = +fFDADist - 2.5; // ns
44  float fFDABBupper = +fFDADist + 2.5; // ns
45  float fFDABGlower = -fFDADist - 4.0; // ns
46  float fFDABGupper = -fFDADist + 4.0; // ns
47 
48  float fFDCBBlower = +fFDCDist - 1.5; // ns
49  float fFDCBBupper = +fFDCDist + 1.5; // ns
50  float fFDCBGlower = -fFDCDist - 2.0; // ns
51  float fFDCBGupper = -fFDCDist + 2.0; // ns
52 
53  float fZNABBlower = -2.0; // ns
54  float fZNABBupper = 2.0; // ns
55  float fZNCBBlower = -2.0; // ns
56  float fZNCBBupper = 2.0; // ns
57 };

113  // read ClassNameToIndexMap from ccdb

117  pClassNameToIndexMap = ccdb.retrieveFromTFileAny<map<string, int>>("Trigger/ClassNameToIndexMap", metadata, run);
118 
119  LOGF(debug, "List of trigger classes");
120  for (auto& cl : *pClassNameToIndexMap) {
121  LOGF(debug, "class %02d %s", cl.second, cl.first);

123 
124  // TODO read aliases from CCDB
125  createAliases();
126 
127  LOGF(debug, "Fill map of alias-to-class-indices");
128  for (auto& al : mAliases) {
129  LOGF(debug, "alias classes: %s", al.second.data());
130  TObjArray* tokens = TString(al.second).Tokenize(",");
131  for (int iClasses = 0; iClasses < tokens->GetEntriesFast(); iClasses++) {
132  string className = tokens->At(iClasses)->GetName();
133  int index = (*pClassNameToIndexMap)[className] - 1;
134  if (index < 0 || index > 63)
135  continue;
136  mAliasToClassIds[al.first].push_back(index);
137  }
138  delete tokens;
139  }

144  void process(aod::Collision const& collision, aod::BCs const& bcs, aod::Zdcs const& zdcs, aod::Run2V0s const& vzeros, aod::FDDs const& fdds)
145  {
146  LOGF(debug, "Starting new event");
147  // CTP info
148  uint64_t triggerMask = collision.bc().triggerMask();
149  LOGF(debug, "triggerMask=%llu", triggerMask);
150  // fill fired aliases
151  int32_t alias[nAliases] = {0};
152  for (auto& al : mAliasToClassIds) {
153  for (auto& classIndex : al.second) {
154  alias[al.first] |= (triggerMask & (1ull << classIndex)) > 0;
155  }
156  }
157 
158  // for (int i=0;i<64;i++) printf("%i",(triggerMask & (1ull << i)) > 0); printf("\n");
159  // for (int i=0;i<nAliases;i++) printf("%i ",alias[i]); printf("\n");
160 
161  // ZDC info
162  auto zdc = getZdc(collision.bc(), zdcs);
163  float timeZNA = zdc.timeZNA();
164  float timeZNC = zdc.timeZNC();
165  // VZERO info
166  auto vzero = getVZero(collision.bc(), vzeros);
167  float timeV0A = vzero.timeA();
168  float timeV0C = vzero.timeC();
169  // FDD info
170  auto fdd = getFDD(collision.bc(), fdds);
171  float timeFDA = fdd.timeA();
172  float timeFDC = fdd.timeC();
173 
174  LOGF(debug, "timeZNA=%f timeZNC=%f", timeZNA, timeZNC);
175  LOGF(debug, "timeV0A=%f timeV0C=%f", timeV0A, timeV0C);
176  LOGF(debug, "timeFDA=%f timeFDC=%f", timeFDA, timeFDC);
177 
178  bool bbZNA = timeZNA > par.fZNABBlower && timeZNA < par.fZNABBupper;
179  bool bbZNC = timeZNC > par.fZNCBBlower && timeZNC < par.fZNCBBupper;
180  bool bbV0A = timeV0A > par.fV0ABBlower && timeV0A < par.fV0ABBupper;
181  bool bbV0C = timeV0C > par.fV0CBBlower && timeV0C < par.fV0CBBupper;
182  bool bgV0A = timeV0A > par.fV0ABGlower && timeV0A < par.fV0ABGupper;
183  bool bgV0C = timeV0C > par.fV0CBGlower && timeV0C < par.fV0CBGupper;
184  bool bbFDA = timeFDA > par.fFDABBlower && timeFDA < par.fFDABBupper;
185  bool bbFDC = timeFDC > par.fFDCBBlower && timeFDC < par.fFDCBBupper;
186  bool bgFDA = timeFDA > par.fFDABGlower && timeFDA < par.fFDABGupper;
187  bool bgFDC = timeFDC > par.fFDCBGlower && timeFDC < par.fFDCBGupper;
188 
189  // Fill event selection columns
190  evsel(alias, bbV0A, bbV0C, bgV0A, bgV0C, bbZNA, bbZNC, bbFDA, bbFDC, bgFDA, bgFDC);
191  }
