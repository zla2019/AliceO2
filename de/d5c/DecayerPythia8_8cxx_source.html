31 void DecayerPythia8::Init()
32 {
33 
35  mPythia.readString("ProcessLevel:all off");
36 
38  auto& param = DecayerPythia8Param::Instance();
39  LOG(INFO) << "Init \'DecayerPythia8\' with following parameters";
40  LOG(INFO) << param;
41  if (!param.config.empty()) {
42  std::stringstream ss(param.config);
43  std::string config;
44  while (getline(ss, config, ' ')) {
45  config = gSystem->ExpandPathName(config.c_str());
46  if (!mPythia.readFile(config, true)) {
47  LOG(FATAL) << "Failed to init \'DecayerPythia8\': problems with configuration file "

49  return;
50  }
51  }
52  }
53 
55  mVerbose = param.verbose;
56 
58  if (!mPythia.init()) {
59  LOG(FATAL) << "Failed to init \'DecayerPythia8\': init returned with error";
60  return;
61  }
62 }

66 void DecayerPythia8::Decay(Int_t pdg, TLorentzVector* lv)
67 {
68  auto mayDecay = mPythia.particleData.mayDecay(pdg); // save mayDecay status
69  mPythia.particleData.mayDecay(pdg, true); // force decay
70 
71  mPythia.event.clear();
72  mPythia.event.append(pdg, 11, 0, 0, lv->Px(), lv->Py(), lv->Pz(), lv->E(), lv->M());
73  mPythia.moreDecays();
74  if (mVerbose)
75  mPythia.event.list();
76 
77  mPythia.particleData.mayDecay(pdg, mayDecay); // restore mayDecay status
78 }

82 Int_t DecayerPythia8::ImportParticles(TClonesArray* particles)
83 {
84  TClonesArray& ca = *particles;
85  ca.Clear();
86 
87  auto nParticles = mPythia.event.size();
88  for (Int_t iparticle = 0; iparticle < nParticles; iparticle++) {
89  // Do not import the decayed particle - start loop from 1
90  auto particle = mPythia.event[iparticle];
91  auto pdg = particle.id();
92  auto st = particle.isFinal();
93  auto px = particle.px();
94  auto py = particle.py();
95  auto pz = particle.pz();
96  auto et = particle.e();
97  auto vx = particle.xProd() * 0.1;
98  auto vy = particle.yProd() * 0.1;
99  auto vz = particle.zProd() * 0.1;
100  auto vt = particle.tProd() * 3.3356410e-12;
101  auto m1 = particle.mother1();
102  auto m2 = particle.mother2();
103  auto d1 = particle.daughter1();
104  auto d2 = particle.daughter2();
105 
106  new (ca[iparticle]) TParticle(pdg, st, m1, m2, d1, d2, px, py, pz, et, vx, vy, vz, vt);
107  }
108 
109  return ca.GetEntries();
110 }
