157 //______________________________________________
158 inline float GRPObject::getBeamZ2A(beamDirection b) const
159 {
160  // Z/A of beam 0 or 1
161  int a = getBeamA(b);
162  return a ? getBeamZ(b) / static_cast<float>(a) : 0.f;
163 }
