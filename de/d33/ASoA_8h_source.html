117 template <typename T, typename ChunkingPolicy = Chunked>
118 class ColumnIterator : ChunkingPolicy
119 {
120  static constexpr char SCALE_FACTOR = std::is_same_v<std::decay_t<T>, bool> ? 3 : 0;
121 
122  public:
127  ColumnIterator(arrow::ChunkedArray const* column)
128  : mColumn{column},
129  mCurrentPos{nullptr},
130  mFirstIndex{0},
131  mCurrentChunk{0}
132  {
133  auto array = getCurrentArray();
134  mCurrent = reinterpret_cast<T const*>(array->values()->data()) + array->offset();
135  mLast = mCurrent + array->length();
136  }
137 
138  ColumnIterator() = default;
139  ColumnIterator(ColumnIterator<T, ChunkingPolicy> const&) = default;
140  ColumnIterator<T, ChunkingPolicy>& operator=(ColumnIterator<T, ChunkingPolicy> const&) = default;
141 
142  ColumnIterator(ColumnIterator<T, ChunkingPolicy>&&) = default;
143  ColumnIterator<T, ChunkingPolicy>& operator=(ColumnIterator<T, ChunkingPolicy>&&) = default;
144 
146  void nextChunk() const
147  {
148  auto previousArray = getCurrentArray();
149  mFirstIndex += previousArray->length();
150 
151  mCurrentChunk++;
152  auto array = getCurrentArray();
153  mCurrent = reinterpret_cast<T const*>(array->values()->data()) + array->offset() - (mFirstIndex >> SCALE_FACTOR);
154  mLast = mCurrent + array->length() + (mFirstIndex >> SCALE_FACTOR);
155  }
156 
157  void prevChunk() const
158  {
159  auto previousArray = getCurrentArray();
160  mFirstIndex -= previousArray->length();
161 
162  mCurrentChunk--;
163  auto array = getCurrentArray();
164  mCurrent = reinterpret_cast<T const*>(array->values()->data()) + array->offset() - (mFirstIndex >> SCALE_FACTOR);
165  mLast = mCurrent + array->length() + (mFirstIndex >> SCALE_FACTOR);
166  }
167 
168  void moveToChunk(int chunk)
169  {
170  if (mCurrentChunk < chunk) {
171  while (mCurrentChunk != chunk) {
172  nextChunk();
173  }
174  } else {
175  while (mCurrentChunk != chunk) {
176  prevChunk();
177  }
178  }
179  }
180 
182  void moveToEnd()
183  {
184  mCurrentChunk = mColumn->num_chunks() - 1;
185  auto array = getCurrentArray();
186  mFirstIndex = mColumn->length() - array->length();
187  mCurrent = reinterpret_cast<T const*>(array->values()->data()) + array->offset() - (mFirstIndex >> SCALE_FACTOR);
188  mLast = mCurrent + array->length() + (mFirstIndex >> SCALE_FACTOR);
189  }
190 
191  decltype(auto) operator*() const
192  {
193  if constexpr (ChunkingPolicy::chunked) {
194  if (O2_BUILTIN_UNLIKELY(((mCurrent + (*mCurrentPos >> SCALE_FACTOR)) >= mLast))) {
195  nextChunk();
196  }
197  }
198  if constexpr (std::is_same_v<bool, std::decay_t<T>>) {
199  // FIXME: notice that due to the type punning we cannot simply convert the
200  // masked char to a bool, because it's undefined behavior.
201  // FIXME: check if shifting the masked bit to the first position is better than != 0
202  return (*((char*)mCurrent + (*mCurrentPos >> SCALE_FACTOR)) & (1 << (*mCurrentPos & 0x7))) != 0;
203  } else {
204  return *(mCurrent + (*mCurrentPos >> SCALE_FACTOR));
205  }
206  }
207 
208  // Move to the chunk which containts element pos
209  ColumnIterator<T>& moveToPos()
210  {
211  // If we get outside range of the current chunk, go to the next.
212  if constexpr (ChunkingPolicy::chunked) {
213  while (O2_BUILTIN_UNLIKELY((mCurrent + (*mCurrentPos >> SCALE_FACTOR)) >= mLast)) {
214  nextChunk();
215  }
216  }
217  return *this;
218  }
219 
220  // Move to the chunk which containts element pos
221  ColumnIterator<T>& checkNextChunk()
222  {
223  if constexpr (ChunkingPolicy::chunked) {
224  if (O2_BUILTIN_LIKELY((mCurrent + (*mCurrentPos >> SCALE_FACTOR)) <= mLast)) {
225  return *this;
226  }
227  nextChunk();
228  }
229  return *this;
230  }
231 
232  mutable T const* mCurrent;
233  int64_t const* mCurrentPos;
234  mutable T const* mLast;
235  arrow::ChunkedArray const* mColumn;
236  mutable int mFirstIndex;
237  mutable int mCurrentChunk;
238 
239  private:
241  auto getCurrentArray() const
242  {
243  std::shared_ptr<arrow::Array> chunkToUse = mColumn->chunk(mCurrentChunk);
244  if constexpr (std::is_same_v<arrow_array_for_t<T>, arrow::FixedSizeListArray>) {
245  chunkToUse = std::dynamic_pointer_cast<arrow::FixedSizeListArray>(chunkToUse)->values();
246  return std::static_pointer_cast<arrow_array_for_t<value_for_t<T>>>(chunkToUse);
247  } else {
248  return std::static_pointer_cast<arrow_array_for_t<T>>(chunkToUse);
249  }
250  }
251 };
